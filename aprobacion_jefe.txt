<script>
var _eventoId = '<?= eventoId ?>';
var _token = '<?= token ?>';
var _registros = [];
var _decisions = {};
var _yaProcesado = false;

var ESTADO_BADGES = {
  'SELECCIONADO': { cls: 'ap-badge-sel', label: 'Pendiente' },
  'PRECONFIRMADO_JEFE': { cls: 'ap-badge-pre', label: 'En proceso' },
  'RECHAZADO': { cls: 'ap-badge-rej', label: 'Rechazado' },
  'CITADO': { cls: 'ap-badge-cit', label: 'En proceso' },
  'CONFIRMADO': { cls: 'ap-badge-conf', label: 'Confirmado' }
};

function init() {
  if (!_eventoId || !_token) { showError('Enlace invalido. Faltan parametros necesarios.'); return; }
  google.script.run
    .withSuccessHandler(onDataLoaded)
    .withFailureHandler(function(e) { showError('Error al cargar datos: ' + (e.message || e)); })
    .getRegistrosParaAprobacion(_eventoId, _token);
}

function onDataLoaded(res) {
  if (!res || !res.success) { showError(res ? res.error : 'Sin respuesta del servidor'); return; }
  _registros = res.registros || [];
  _yaProcesado = res.yaProcesado || false;
  if (!_registros.length) {
    showMsg('No se encontraron participantes de su equipo para este curso.', 'error');
    return;
  }
  // Pre-populate decisions from existing states
  for (var i = 0; i < _registros.length; i++) {
    var r = _registros[i];
    if (r.estado === 'PRECONFIRMADO_JEFE' || r.estado === 'CONFIRMADO') _decisions[r.regId] = 'APROBAR';
    else if (r.estado === 'RECHAZADO') _decisions[r.regId] = 'RECHAZAR';
  }
  renderPage(res);
}

function renderPage(res) {
  var ev = res.evento || {};
  var h = '';
  // Course info card
  h += '<div class="ap-card">';
  h += '<div class="ap-card-hd"><span style="font-size:18px;">&#128218;</span><h2>Informacion del Curso</h2></div>';
  h += '<div class="ap-card-bd"><div class="ap-info-grid">';
  h += '<div class="ap-info-item"><label>Curso</label><span>' + (ev.sigla || '-') + '</span></div>';
  h += '<div class="ap-info-item"><label>Fecha</label><span>' + (ev.fecha || '-') + '</span></div>';
  h += '<div class="ap-info-item"><label>Instructor</label><span>' + (ev.instructor || '-') + '</span></div>';
  h += '<div class="ap-info-item"><label>Sala</label><span>' + (ev.sala || '-') + '</span></div>';
  h += '</div></div></div>';

  // Status alert
  if (_yaProcesado) {
    h += '<div class="ap-alert ap-alert-warn">';
    h += '<span style="font-size:16px;">&#9888;</span>';
    h += '<div><strong>Este curso ya fue procesado.</strong><br/>Puede revisar sus decisiones anteriores y actualizarlas si lo necesita. Los cambios se guardaran al presionar "Confirmar Decisiones".</div>';
    h += '</div>';
  } else {
    h += '<div class="ap-alert ap-alert-info">';
    h += '<span style="font-size:16px;">&#8505;</span>';
    h += '<div><strong>Instrucciones:</strong> Para cada participante, seleccione <strong>Aprobar</strong> o <strong>Rechazar</strong>. Opcionalmente agregue un comentario. Luego presione "Confirmar Decisiones".</div>';
    h += '</div>';
  }

  // Participants table
  h += '<div class="ap-card">';
  h += '<div class="ap-card-hd"><span style="font-size:18px;">&#128101;</span><h2>' + _registros.length + ' Participante' + (_registros.length > 1 ? 's' : '') + '</h2>';
  if (res.pendientes > 0 && res.procesados > 0) {
    h += '<span style="font-size:11px;color:#94a3b8;margin-left:auto;">' + res.pendientes + ' pendiente' + (res.pendientes > 1 ? 's' : '') + ', ' + res.procesados + ' procesado' + (res.procesados > 1 ? 's' : '') + '</span>';
  }
  h += '</div>';
  h += '<div style="overflow-x:auto;"><table class="ap-table"><thead><tr>';
  h += '<th>N</th><th>BP</th><th>Nombre</th><th>Area</th><th>Estado</th><th>Decision</th>';
  h += '</tr></thead><tbody>';

  for (var i = 0; i < _registros.length; i++) {
    var r = _registros[i];
    var badge = ESTADO_BADGES[r.estado] || { cls: '', label: r.estado };
    var isEditable = r.editable;
    var prevDec = _decisions[r.regId] || '';

    h += '<tr id="row-' + r.regId + '"' + (r.estado === 'PRECONFIRMADO_JEFE' || r.estado === 'CONFIRMADO' ? ' style="background:#f0fdf4;"' : r.estado === 'RECHAZADO' ? ' style="background:#fef2f2;"' : '') + '>';
    h += '<td>' + (i + 1) + '</td>';
    h += '<td class="ap-bp">' + r.bp + '</td>';
    h += '<td class="ap-name">' + r.nombre + '</td>';
    h += '<td class="ap-area">' + r.area + '</td>';
    h += '<td><span class="ap-badge ' + badge.cls + '">' + badge.label + '</span></td>';
    h += '<td>';
    if (isEditable) {
      h += '<div class="ap-radio-group">';
      h += '<button class="ap-radio-btn approve' + (prevDec === 'APROBAR' ? ' active' : '') + '" data-regid="' + r.regId + '" data-dec="APROBAR" onclick="setDecision(this)">&#10003; Aprobar</button>';
      h += '<button class="ap-radio-btn reject' + (prevDec === 'RECHAZAR' ? ' active' : '') + '" data-regid="' + r.regId + '" data-dec="RECHAZAR" onclick="setDecision(this)">&#10007; Rechazar</button>';
      h += '</div>';
      var prevComment = r.comentarioJefe || r.nota || '';
      h += '<textarea class="ap-comment" id="comment-' + r.regId + '" rows="1" placeholder="Comentario (opcional)...">' + (prevComment && r.estado !== 'SELECCIONADO' ? prevComment : '') + '</textarea>';
      if (prevComment && r.estado !== 'SELECCIONADO') {
        h += '<div class="ap-nota-prev">Nota anterior: ' + prevComment + '</div>';
      }
    } else {
      h += '<span style="font-size:11px;color:#94a3b8;">No editable (estado avanzado)</span>';
    }
    h += '</td></tr>';
  }
  h += '</tbody></table></div>';

  // Bulk bar
  var hasEditable = _registros.some(function(r) { return r.editable; });
  if (hasEditable) {
    h += '<div class="ap-bulk-bar">';
    h += '<button class="ap-btn ap-btn-secondary" onclick="bulkAll(\'APROBAR\')">Aprobar todos</button>';
    h += '<button class="ap-btn ap-btn-secondary" onclick="bulkAll(\'RECHAZAR\')" style="color:#dc2626;">Rechazar todos</button>';
    h += '<div style="flex:1;"></div>';
    h += '<button class="ap-btn ap-btn-primary" id="btn-submit" onclick="submitDecisions()">' + (_yaProcesado ? 'Actualizar Decisiones' : 'Confirmar Decisiones') + '</button>';
    h += '</div>';
  }
  h += '</div>';
  h += '<div id="ap-result" style="margin-top:16px;"></div>';
  document.getElementById('ap-content').innerHTML = h;
}

function setDecision(btn) {
  var regId = btn.getAttribute('data-regid');
  var dec = btn.getAttribute('data-dec');
  _decisions[regId] = dec;
  var btns = btn.parentNode.querySelectorAll('.ap-radio-btn');
  for (var i = 0; i < btns.length; i++) btns[i].classList.remove('active');
  btn.classList.add('active');
  var row = document.getElementById('row-' + regId);
  if (row) row.style.background = dec === 'APROBAR' ? '#f0fdf4' : dec === 'RECHAZAR' ? '#fef2f2' : '';
}

function bulkAll(dec) {
  var btns = document.querySelectorAll('.ap-radio-btn[data-dec="' + dec + '"]');
  for (var i = 0; i < btns.length; i++) {
    if (!btns[i].disabled) setDecision(btns[i]);
  }
}

function submitDecisions() {
  var decisiones = [];
  var editables = _registros.filter(function(r) { return r.editable; });
  for (var i = 0; i < editables.length; i++) {
    var regId = editables[i].regId;
    var dec = _decisions[regId];
    if (!dec) { appAlert('Seleccione una decision para: ' + editables[i].nombre); return; }
    var comment = document.getElementById('comment-' + regId);
    decisiones.push({ regId: regId, decision: dec, comentario: comment ? comment.value.trim() : '' });
  }
  if (!decisiones.length) { appAlert('No hay participantes editables.'); return; }
  var btn = document.getElementById('btn-submit');
  btn.disabled = true; btn.textContent = 'Enviando...';
  google.script.run
    .withSuccessHandler(function(res) {
      if (res && res.success) {
        var msg = '';
        if (res.aprobados > 0) msg += res.aprobados + ' confirmado' + (res.aprobados > 1 ? 's' : '');
        if (res.rechazados > 0) msg += (msg ? ', ' : '') + res.rechazados + ' rechazado' + (res.rechazados > 1 ? 's' : '');
        if (res.sinCambio > 0) msg += (msg ? ', ' : '') + res.sinCambio + ' sin cambios';
        showMsg('Decisiones procesadas exitosamente. ' + msg + '.', 'success');
      } else {
        showError('Error: ' + (res ? res.error : 'Sin respuesta'));
        btn.disabled = false; btn.textContent = 'Confirmar Decisiones';
      }
    })
    .withFailureHandler(function(e) {
      showError('Error: ' + (e.message || e));
      btn.disabled = false; btn.textContent = 'Confirmar Decisiones';
    })
    .procesarAprobacionJefe(_eventoId, _token, decisiones);
}

function showMsg(msg, tipo) {
  document.getElementById('ap-content').innerHTML = '<div class="ap-msg ' + tipo + '"><p style="font-size:24px;margin-bottom:8px;">' + (tipo === 'success' ? '&#10004;' : '&#9888;') + '</p><p>' + msg + '</p><p style="margin-top:12px;font-size:12px;color:#64748b;">Puede cerrar esta ventana.</p></div>';
}
function showError(msg) {
  document.getElementById('ap-content').innerHTML = '<div class="ap-msg error"><p style="font-size:24px;margin-bottom:8px;">&#9888;</p><p>' + msg + '</p></div>';
}
init();
</script>
