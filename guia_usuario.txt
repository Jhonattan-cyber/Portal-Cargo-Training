function mostrarSeccionGuia(secId) {
  var secs = document.querySelectorAll('[id^="guia-sec-"]');
  for (var i = 0; i < secs.length; i++) secs[i].style.display = 'none';
  var target = document.getElementById('guia-sec-' + secId);
  if (target) target.style.display = 'block';
  var btns = document.querySelectorAll('.guia-tab-btn');
  for (var j = 0; j < btns.length; j++) {
    btns[j].getAttribute('data-guia-tab') === secId ? btns[j].classList.add('active') : btns[j].classList.remove('active');
  }
}

function imprimirGuia() {
  var secs = document.querySelectorAll('[id^="guia-sec-"]');
  for (var i = 0; i < secs.length; i++) secs[i].style.display = 'block';
  setTimeout(function() { window.print(); }, 200);
}
