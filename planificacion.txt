// V4-UX: Formato de fecha corta para display
function _fmtFechaCorta(val) {
  if (!val) return '';
  try {
    var d = new Date(String(val));
    if (isNaN(d.getTime())) return String(val).substring(0, 10);
    var meses = ['Ene','Feb','Mar','Abr','May','Jun','Jul','Ago','Sep','Oct','Nov','Dic'];
    var dd = d.getDate();
    var mm = meses[d.getMonth()];
    var yy = d.getFullYear();
    var hh = d.getHours();
    var mi = d.getMinutes();
    var ampm = hh >= 12 ? 'PM' : 'AM';
    hh = hh % 12 || 12;
    return dd + ' ' + mm + ' ' + yy + ' – ' + hh + ':' + (mi < 10 ? '0' : '') + mi + ' ' + ampm;
  } catch(e) { return String(val).substring(0, 10); }
}

function _ssCreate(wrapperId, options, placeholder, onChange) {
  var wrap = document.getElementById(wrapperId);
  if (!wrap) return;
  wrap.classList.add('ss-wrap');
  var curVal = '';
  var inp = document.createElement('input');
  inp.className = 'ss-input';
  inp.placeholder = placeholder || 'Buscar...';
  inp.setAttribute('data-testid', wrapperId + '-input');
  var dd = document.createElement('div');
  dd.className = 'ss-dropdown';
  dd.setAttribute('data-testid', wrapperId + '-dropdown');
  wrap.innerHTML = '';
  wrap.appendChild(inp);
  wrap.appendChild(dd);

  function render(filter) {
    var f = (filter || '').toLowerCase();
    var h = '<div class="ss-opt' + (!curVal ? ' active' : '') + '" data-val="">' + placeholder + '</div>';
    for (var i = 0; i < options.length; i++) {
      var o = options[i];
      if (f && o.toLowerCase().indexOf(f) < 0) continue;
      h += '<div class="ss-opt' + (o === curVal ? ' active' : '') + '" data-val="' + o + '">' + o + '</div>';
    }
    dd.innerHTML = h;
  }

  inp.addEventListener('focus', function() { render(inp.value); dd.classList.add('open'); });
  inp.addEventListener('input', function() { render(inp.value); dd.classList.add('open'); });
  inp.addEventListener('blur', function() { setTimeout(function() { dd.classList.remove('open'); }, 180); });

  dd.addEventListener('mousedown', function(e) {
    e.preventDefault();
    var opt = e.target.closest('.ss-opt');
    if (!opt) return;
    curVal = opt.getAttribute('data-val') || '';
    inp.value = curVal;
    dd.classList.remove('open');
    if (onChange) onChange(curVal);
  });

  // Allow clearing with empty input
  inp.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') {
      dd.classList.remove('open');
      // If typed text matches an option exactly, select it
      var typed = inp.value.trim();
      var found = false;
      for (var i = 0; i < options.length; i++) {
        if (options[i].toLowerCase() === typed.toLowerCase()) { curVal = options[i]; inp.value = curVal; found = true; break; }
      }
      if (!typed) { curVal = ''; }
      if (onChange) onChange(curVal);
    }
    if (e.key === 'Escape') { dd.classList.remove('open'); }
  });

  return {
    setValue: function(v) { curVal = v; inp.value = v; },
    getValue: function() { return curVal; },
    setOptions: function(newOpts) { options = newOpts; render(''); }
  };
}

function _planConfirmOk() {
  var ui = _planModalGet();
  if (ui.overlay) ui.overlay.style.display = 'none';

  if (_planModalState.mode === 'prompt') {
    var val = (ui.input ? ui.input.value : '').trim();

      if (ui.inputError) {
        ui.inputError.textContent = 'Este campo es obligatorio.';
        ui.inputError.style.display = 'block';
      }
      if (ui.input) ui.input.focus();
      if (ui.overlay) ui.overlay.style.display = 'flex';
      return;
    }
    var cbPrompt = _planModalState.onConfirm;
    _planModalState.onConfirm = null;
    if (typeof cbPrompt === 'function') cbPrompt(val);

  }

    var cb = _planModalState.onConfirm;
    _planModalState.onConfirm = null;
    cb();
    return;
  }

  _planModalState.onConfirm = null;
}

    }).getEventosCalendario({ mesAnio: nextKey, sigla: siglaVal, instructor: instrVal });

function loadPlanEvents(forceRefresh) {
  var cal = document.getElementById('plan-cal');
  var mesAnio = _pAnio + '-' + String(_pMes + 1).padStart(2, '0');

  var cacheKey = mesAnio + '|' + siglaVal + '|' + instrVal;
  
  // V102-PERF: Verificar cache local antes de ir al servidor
  if (!forceRefresh && _eventsCache[cacheKey]) {
    var cached = _eventsCache[cacheKey];
    if (Date.now() - cached.timestamp < _eventsCacheTTL) {
      _pEventos = cached.eventos;
      document.getElementById('plan-ev-count').textContent = _pEventos.length + ' evento' + (_pEventos.length !== 1 ? 's' : '');
      _instrColorMap = {};
      _instrColorIdx = 0;
      _updateFilters();
      _renderCal();
      return;
    }
  }
  
  cal.innerHTML = '<div style="display:flex;align-items:center;justify-content:center;height:300px;gap:10px;"><div class="plan-spin"></div><span style="font-size:12px;color:#94a3b8;">Cargando eventos...</span></div>';
  // V6: Reset instructor color map per reload for consistency
  _instrColorMap = {};
  _instrColorIdx = 0;
  google.script.run
    .withSuccessHandler(function(res) {
      if (!res || !res.success) {
        cal.innerHTML = '<div style="padding:40px;text-align:center;"><i class="ph ph-warning" style="font-size:28px;color:#dc2626;"></i><p style="color:#dc2626;font-size:12px;margin-top:8px;">' + (res ? res.error : 'Error desconocido') + '</p></div>';
        return;
      }
      _pEventos = res.eventos || [];
      // V102-PERF: Guardar en cache local
      _eventsCache[cacheKey] = { eventos: _pEventos, timestamp: Date.now() };
      document.getElementById('plan-ev-count').textContent = _pEventos.length + ' evento' + (_pEventos.length !== 1 ? 's' : '');
      _updateFilters();
      _renderCal();
    })
    .withFailureHandler(function(e) {
      cal.innerHTML = '<div style="padding:40px;text-align:center;"><i class="ph ph-warning" style="font-size:28px;color:#dc2626;"></i><p style="color:#dc2626;font-size:12px;margin-top:8px;">' + (e.message || e) + '</p></div>';
    })
    .getEventosCalendario({ mesAnio: mesAnio, sigla: siglaVal, instructor: instrVal });
}

function _updateFilters() {
  var siglas = {}, instrs = {};
  for (var i = 0; i < _pEventos.length; i++) {
    if (_pEventos[i].sigla) siglas[_pEventos[i].sigla] = true;
    if (_pEventos[i].instructor) instrs[_pEventos[i].instructor] = true;
  }
  var siglaArr = Object.keys(siglas).sort();
  var instrArr = Object.keys(instrs).sort();

  // V6: Usar searchable selects
  if (!_ssPlanSigla) {
    _ssPlanSigla = _ssCreate('plan-ss-sigla', siglaArr, 'Sigla...', function() { loadPlanEvents(); });
  } else {
    _ssPlanSigla.setOptions(siglaArr);
  }
  if (!_ssPlanInstr) {
    _ssPlanInstr = _ssCreate('plan-ss-instr', instrArr, 'Instructor...', function() { loadPlanEvents(); });
  } else {
    _ssPlanInstr.setOptions(instrArr);
  }
}

// === RENDER CALENDAR ===
function _renderCal() {
  var cal = document.getElementById('plan-cal');
  var totalDias = new Date(_pAnio, _pMes + 1, 0).getDate();
  var evPorDia = {};
  
  // V102: Pre-calcular indicador de día (D1, D2, D3...) para cursos multi-día
  // Agrupa por SIGLA+INSTRUCTOR, ordena por fecha, y asigna número secuencial
  _multiDiaMap = {}; // Reset global
  var _gruposCurso = {}; // key: sigla|instructor → [{id, fecha}]
  for (var i = 0; i < _pEventos.length; i++) {
    var ev = _pEventos[i];
    var d = parseInt(ev.fecha.split('-')[2], 10);
    if (!evPorDia[d]) evPorDia[d] = [];
    evPorDia[d].push(ev);
    // Agrupar si no tiene día manual
    if (!ev.dia) {
      var gKey = (ev.sigla || '').toUpperCase() + '|' + (ev.instructor || '').toUpperCase();
      if (!_gruposCurso[gKey]) _gruposCurso[gKey] = [];
      _gruposCurso[gKey].push({ id: ev.id, fecha: ev.fecha });
    }
  }
  // Calcular día secuencial para grupos con >1 evento
  var gKeys = Object.keys(_gruposCurso);
  for (var g = 0; g < gKeys.length; g++) {
    var grupo = _gruposCurso[gKeys[g]];
    if (grupo.length <= 1) continue; // No es multi-día
    grupo.sort(function(a, b) { return a.fecha < b.fecha ? -1 : a.fecha > b.fecha ? 1 : 0; });
    for (var dIdx = 0; dIdx < grupo.length; dIdx++) {
      _multiDiaMap[grupo[dIdx].id] = (dIdx + 1);
    }
  }
  
  var hoy = new Date();
  var esHoyMes = (hoy.getMonth() === _pMes && hoy.getFullYear() === _pAnio);
  var diaHoy = hoy.getDate();

  var h = '<div class="plan-grid">';
  for (var w = 0; w < 5; w++) h += '<div class="plan-weekday">' + _DIAS5[w] + '</div>';

  // Find first weekday
  var primerHabil = 1;
  while (primerHabil <= totalDias && (new Date(_pAnio, _pMes, primerHabil).getDay() % 6 === 0)) primerHabil++;
  if (primerHabil <= totalDias) {
    var off = new Date(_pAnio, _pMes, primerHabil).getDay() - 1;
    for (var s = 0; s < off; s++) h += '<div class="plan-day-cell is-empty"></div>';
  }

  for (var dia = 1; dia <= totalDias; dia++) {
    var dow = new Date(_pAnio, _pMes, dia).getDay();
    if (dow === 0 || dow === 6) continue;
    var esHoy = esHoyMes && dia === diaHoy;
    var evts = evPorDia[dia] || [];
    var clk = evts.length === 1 ? 'openPlanEvt(\'' + evts[0].id + '\')' : (evts.length > 1 ? 'openDayModal(' + dia + ')' : '');

    h += '<div class="plan-day-cell' + (esHoy ? ' is-today' : '') + '"' + (clk ? ' onclick="' + clk + '"' : '') + '>';
    h += '<div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:3px;">';
    h += '<span class="plan-day-num' + (esHoy ? ' today-circle' : '') + '">' + dia + '</span>';
    if (evts.length > 0) h += '<span class="plan-day-badge">' + evts.length + '</span>';
    h += '</div>';

    // V3.2: Mostrar TODOS los eventos sin truncar (igual que Calendario)
    var maxC = 999;
    for (var e = 0; e < Math.min(evts.length, maxC); e++) {
      var ev = evts[e];
      var ic = _getInstrColor(ev.instructor);
      h += '<div class="plan-chip" style="background:' + ic.bg + ';border-left-color:' + ic.border + ';color:' + ic.text + ';" onclick="event.stopPropagation();openPlanEvt(\'' + ev.id + '\')">';
      // Instructor como header
      if (ev.instructor) {
        var instrParts = ev.instructor.split(' ');
        var instrShort = instrParts[0] + (instrParts.length > 1 ? ' ' + instrParts[1].charAt(0) + '.' : '');
        h += '<div class="plan-chip-instr">' + instrShort + '</div>';
      }
      // Sigla en badge + indicador de dia + titulo completo
      h += '<div class="plan-chip-title">';
      h += '<span class="plan-chip-sigla">' + (ev.sigla || '?') + '</span>';
      var _diaIndicador = ev.dia || _multiDiaMap[ev.id] || '';
      if (_diaIndicador) h += '<span class="plan-chip-dia">D' + _diaIndicador + '</span>';
      if (ev.titulo) h += '<span class="plan-chip-curso">' + ev.titulo + '</span>';
      h += '</div>';
      // Sala + horario
      h += '<div class="plan-chip-meta">';
      if (ev.sala) h += ev.sala + ' ';
      if (ev.horaInicio) {
        h += ev.horaInicio;
        if (ev.horaFin) h += '-' + ev.horaFin;
      }
      h += '</div></div>';
    }
    // V3.2: Sin truncacion — se muestran todos los eventos
    h += '</div>';
  }
  h += '</div>';

  // V6: Leyenda de colores por instructor
  var instrKeys = Object.keys(_instrColorMap);
  if (instrKeys.length > 0) {
    h += '<div style="display:flex;gap:12px;flex-wrap:wrap;padding:10px 0 2px;border-top:1px solid #f1f5f9;margin-top:6px;" data-testid="instr-legend">';
    for (var il = 0; il < instrKeys.length; il++) {
      var instrName = instrKeys[il];
      var ic2 = _instrColorMap[instrName];
      // Capitalize first letter
      var displayName = instrName.charAt(0) + instrName.slice(1).toLowerCase();
      h += '<div style="display:flex;align-items:center;gap:4px;font-size:10px;font-weight:600;color:' + ic2.text + ';">';
      h += '<div style="width:10px;height:10px;border-radius:3px;background:' + ic2.bg + ';border:2px solid ' + ic2.border + ';"></div>';
      h += displayName + '</div>';
    }
    h += '</div>';
  }

  cal.innerHTML = h;
}

// === PIPELINE VISUAL (V102 — 3 pasos: Seleccionado → Pre-confirmado → Confirmado) ===
function _renderPipeline(regs) {
  var el = document.getElementById('planM-pipeline');
  if (!el) return;
  if (!regs.length) { el.innerHTML = ''; return; }
  var counts = {};
  var preConfCount = 0;
  // V3.12: Contadores post-curso (sobrescriben el flujo previo cuando aplica)
  var postAprobCount = 0, postReprobCount = 0, postNoAsistCount = 0, postPendEvalCount = 0;
  for (var i = 0; i < regs.length; i++) {
    var est = regs[i].estadoContacto;
    var estOp = regs[i].estadoOperativo || 'PRE_CURSO';
    // Si tiene estado operativo terminal, NO se cuenta en el pipeline previo
    if (estOp === 'APROBADO_PEND_FORM' || estOp === 'FORMALIZADO') { postAprobCount++; continue; }
    if (estOp === 'REPROBADO') { postReprobCount++; continue; }
    if (estOp === 'NO_ASISTIO') { postNoAsistCount++; continue; }
    if (estOp === 'ASISTIO_PEND_EVAL') { postPendEvalCount++; continue; }
    counts[est] = (counts[est] || 0) + 1;
    // V102: Agrupar legacy como PRE_CONFIRMADO
    if (est === 'PRE_CONFIRMADO' || est === 'PRECONFIRMADO_JEFE' || est === 'CITADO' || est === 'NO_CONTACTADO') preConfCount++;
  }
  var h = '<div class="plan-pipe-wrap">';
  // SELECCIONADO
  var selCnt = counts['SELECCIONADO'] || 0;
  var selActive = _prFilterEstado === 'SELECCIONADO';
  h += '<div class="plan-pipe-item" style="background:#f59e0b' + (selActive ? '28' : '12') + ';border:' + (selActive ? '2px solid #f59e0b' : '1px solid #f59e0b25') + ';" onclick="_setPRFilter(\'SELECCIONADO\')" data-testid="plan-pipe-SELECCIONADO">';
  h += '<div class="cnt" style="color:#f59e0b;">' + selCnt + '</div>';
  h += '<div class="lbl" style="color:#f59e0b;">Seleccionado</div></div>';
  h += '<div class="plan-pipe-arrow"><i class="ph ph-caret-right"></i></div>';
  // PRE_CONFIRMADO (V102: incluye legacy)
  var pcActive = _prFilterEstado === 'PRE_CONFIRMADO';
  h += '<div class="plan-pipe-item" style="background:#8b5cf6' + (pcActive ? '28' : '12') + ';border:' + (pcActive ? '2px solid #8b5cf6' : '1px solid #8b5cf625') + ';" onclick="_setPRFilter(\'PRE_CONFIRMADO\')" data-testid="plan-pipe-PRE_CONFIRMADO">';
  h += '<div class="cnt" style="color:#8b5cf6;">' + preConfCount + '</div>';
  h += '<div class="lbl" style="color:#8b5cf6;">Pre-confirmado</div></div>';
  h += '<div class="plan-pipe-arrow"><i class="ph ph-caret-right"></i></div>';
  // CONFIRMADO
  var confCnt = counts['CONFIRMADO'] || 0;
  var confActive = _prFilterEstado === 'CONFIRMADO';
  h += '<div class="plan-pipe-item" style="background:#22c55e' + (confActive ? '28' : '12') + ';border:' + (confActive ? '2px solid #22c55e' : '1px solid #22c55e25') + ';" onclick="_setPRFilter(\'CONFIRMADO\')" data-testid="plan-pipe-CONFIRMADO">';
  h += '<div class="cnt" style="color:#22c55e;">' + confCnt + '</div>';
  h += '<div class="lbl" style="color:#22c55e;">Confirmado</div></div>';
  // RECHAZADO (secondary)
  var rej = counts['RECHAZADO'] || 0;
  if (rej > 0) {
    h += '<div class="plan-pipe-secondary">';
    h += '<span style="color:#ef4444;cursor:pointer;" onclick="_setPRFilter(\'RECHAZADO\')" data-testid="plan-pipe-REJ">' + rej + ' Rechazado' + (rej > 1 ? 's' : '') + '</span>';
    h += '</div>';
  }
  // V3.12: Bloque POST-CURSO (separado visualmente). Refleja lo que el instructor llenó en Calendario.
  var postTotal = postAprobCount + postReprobCount + postNoAsistCount + postPendEvalCount;
  if (postTotal > 0) {
    h += '<div class="plan-pipe-arrow"><i class="ph ph-caret-right"></i></div>';
    h += '<div style="display:flex;align-items:center;gap:6px;flex-wrap:wrap;background:#fafafa;border-radius:10px;padding:6px 10px;border:1px dashed #cbd5e1;">';
    h += '<span style="font-size:9px;font-weight:800;color:#64748b;text-transform:uppercase;letter-spacing:0.5px;">Post-curso</span>';
    if (postAprobCount > 0) h += '<span style="font-size:11px;font-weight:800;color:#22c55e;background:#dcfce7;padding:2px 8px;border-radius:8px;border:1px solid #22c55e;" data-testid="plan-pipe-POSTAPROB">✅ ' + postAprobCount + ' Aprobado' + (postAprobCount > 1 ? 's' : '') + '</span>';
    if (postReprobCount > 0) h += '<span style="font-size:11px;font-weight:800;color:#ef4444;background:#fee2e2;padding:2px 8px;border-radius:8px;border:1px solid #ef4444;" data-testid="plan-pipe-POSTREPROB">🔁 ' + postReprobCount + ' Reprobado' + (postReprobCount > 1 ? 's' : '') + '</span>';
    if (postNoAsistCount > 0) h += '<span style="font-size:11px;font-weight:800;color:#92400e;background:#fef3c7;padding:2px 8px;border-radius:8px;border:1px solid #d97706;" data-testid="plan-pipe-NOASIST">⛔ ' + postNoAsistCount + ' No asistió' + '</span>';
    if (postPendEvalCount > 0) h += '<span style="font-size:11px;font-weight:800;color:#f59e0b;background:#fef9c3;padding:2px 8px;border-radius:8px;border:1px solid #f59e0b;" data-testid="plan-pipe-PENDEVAL">⏳ ' + postPendEvalCount + ' Pend. eval</span>';
    h += '</div>';
  }
  h += '</div>';
  el.innerHTML = h;
}

function _renderStepper(regs) {
  var el = document.getElementById('planM-stepper');
  if (!el) return;
  if (!regs || !regs.length) { el.style.display = 'none'; return; }
  el.style.display = 'flex';
  var counts = {};
  for (var i = 0; i < regs.length; i++) {
    var est = regs[i].estadoContacto;
    counts[est] = (counts[est] || 0) + 1;
  }

    if (counts[STEPPER_STEPS[s].key] > 0) { activeIdx = s; break; }
  }
  var h = '';
  for (var j = 0; j < STEPPER_STEPS.length; j++) {
    var step = STEPPER_STEPS[j];
    var cls = 'plan-step';
    if (j < activeIdx) cls += ' done';
    else if (j === activeIdx) cls += ' active';
    h += '<div class="' + cls + '" data-testid="plan-step-' + step.key + '">';
    if (j < STEPPER_STEPS.length - 1) h += '<div class="plan-step-line"></div>';
    h += '<div class="plan-step-num">' + step.num + '</div>';
    h += '<span class="plan-step-label">' + step.label + '</span>';
    h += '<span class="plan-step-desc">' + step.desc + '</span>';
    h += '</div>';
  }
  el.innerHTML = h;
}

// === RENDER TABLA OPERATIVA (V3 — con decision y comentario) ===
function _renderPRTable() {
  var tbl = document.getElementById('planM-pr-table');
  var regs = _prRegs;
  // V102: Agrupar filtro PRE_CONFIRMADO = nuevo estado + legacy states
  var filtered = regs;
  if (_prFilterEstado) {
    filtered = [];
    for (var f = 0; f < regs.length; f++) {
      var est = regs[f].estadoContacto;
      if (_prFilterEstado === 'PRE_CONFIRMADO') {
        if (est === 'PRE_CONFIRMADO' || est === 'PRECONFIRMADO_JEFE' || est === 'CITADO' || est === 'NO_CONTACTADO') filtered.push(regs[f]);
      } else if (est === _prFilterEstado) {
        filtered.push(regs[f]);
      }
    }
  }
  // V102: Count map con PRE_CONFIRMADO agrupado
  var countMap = {};
  var preConfCount = 0;
  for (var ci = 0; ci < regs.length; ci++) {
    var e = regs[ci].estadoContacto;
    countMap[e] = (countMap[e] || 0) + 1;
    if (e === 'PRE_CONFIRMADO' || e === 'PRECONFIRMADO_JEFE' || e === 'CITADO' || e === 'NO_CONTACTADO') preConfCount++;
  }
  // V102: Filter pills actualizados
  var fh = '<div class="plan-filter-pills">';
  fh += '<button onclick="_setPRFilter(\'\')" class="plan-filter-pill' + (!_prFilterEstado ? ' active' : '') + '" style="' + (!_prFilterEstado ? 'background:var(--pl-blue);color:white;' : '') + '" data-testid="plan-filter-all">Todos (' + regs.length + ')</button>';
  // Seleccionado
  if (countMap['SELECCIONADO']) {
    var isAct = _prFilterEstado === 'SELECCIONADO';
    fh += '<button onclick="_setPRFilter(\'SELECCIONADO\')" class="plan-filter-pill' + (isAct ? ' active' : '') + '" style="' + (isAct ? 'background:#f59e0b;color:white;' : '') + '" data-testid="plan-filter-SELECCIONADO">Seleccionado (' + countMap['SELECCIONADO'] + ')</button>';
  }
  // Pre-confirmado (V102: incluye legacy)

// V106-FIX: Optimistic UI + validación + rechazo con modal
function _onEstadoSel(regId) {
  var sel = document.getElementById('sel-' + regId);
  if (!sel) return;

  var nuevoEstado = sel.value;
  if (!nuevoEstado) return;

  var estadoAnterior = sel.getAttribute('data-prev-value') || '';

  // Buscar fila
  var row = sel.closest('tr');

  function _restaurar() {
    sel.value = estadoAnterior;
    sel.disabled = false;
    sel.style.opacity = '1';
    sel.style.pointerEvents = '';
    if (row) row.style.opacity = '1';
  }

  function _bloquear() {
    sel.disabled = true;
    sel.style.opacity = '.6';
    sel.style.pointerEvents = 'none';

    // feedback visual inmediato
    if (row) {
      row.style.opacity = '.5';
    }
  }

  function _commit(motivo) {
    motivo = (motivo || '').trim();

    _bloquear();
    _planToast('Actualizando...', 'info');

    // OPTIMISTIC UI: cambiar visualmente badge antes del backend
    if (row) {
      var estadoCell = row.querySelector('td:nth-child(5)'); // columna ESTADO
      if (estadoCell) {
        estadoCell.innerHTML = '<span style="font-size:10px;color:#64748b;">Actualizando...</span>';
      }
    }

    google.script.run
      .withSuccessHandler(function(r) {
        if (r && r.success) {
          _planToast('Estado actualizado', 'ok');

          // recarga silenciosa (no bloquea UX)
          setTimeout(function() {
            if (_pEvtActual) loadPR(_pEvtActual.id);
          }, 200);

        } else {
          _restaurar();
          _planToast('Error: ' + (r ? r.error : 'sin respuesta'), 'err');
        }
      })
      .withFailureHandler(function(e) {
        _restaurar();
        _planToast('Error: ' + (e.message || e), 'err');
      })
      .actualizarContacto(regId, nuevoEstado, motivo);
  }

  // =========================
  //   CASO RECHAZO (modal)
  // =========================
  if (nuevoEstado === 'RECHAZADO') {

    window._prRevertEstadoSel = _restaurar;

    _planPromptShow(
      'Motivo del rechazo (obligatorio):',
      function(motivo) {
        window._prRevertEstadoSel = null;
        _commit(motivo);
      },
      {
        title: 'Rechazar colaborador',
        label: 'Motivo del rechazo (obligatorio):',
        placeholder: 'Escribe el motivo del rechazo...',
        confirmText: 'Rechazar',
        iconClass: 'ph ph-x-circle',
        iconColor: '#dc2626',
        iconBg: '#fef2f2',
        confirmBg: 'linear-gradient(135deg,#dc2626 0%,#b91c1c 100%)'
      }
    );

    return;
  }

  // =========================
  //      OTROS ESTADOS
  // =========================
  _commit('');
}

function _confirmarCambio(regId, nuevoEstado, motivo, sel, onErrorRestore) {
  sel = sel || document.getElementById('sel-' + regId);
  if (!sel) return;

  var estadoAnterior = sel.getAttribute('data-prev-value') || '';

  function _restaurarSelect() {
    if (typeof onErrorRestore === 'function') {
      onErrorRestore();
      return;
    }
    sel.value = estadoAnterior;
    sel.setAttribute('data-prev-value', estadoAnterior);
    sel.disabled = false;
    sel.style.opacity = '1';
    sel.style.pointerEvents = '';
  }

  sel.disabled = true;
  sel.style.opacity = '.6';
  sel.style.pointerEvents = 'none';
  _planToast('Actualizando...', 'info');

  google.script.run
    .withSuccessHandler(function(r) {
      if (r && r.success) {
        sel.setAttribute('data-prev-value', nuevoEstado);
        _planToast('Estado actualizado', 'ok');
        if (_pEvtActual) loadPR(_pEvtActual.id);
      } else {
        _restaurarSelect();
        _planToast('Error: ' + (r ? r.error : 'sin respuesta'), 'err');
      }
    })
    .withFailureHandler(function(e) {
      _restaurarSelect();
      _planToast('Error: ' + (e.message || e), 'err');
    })
    .actualizarContacto(regId, nuevoEstado, motivo || '');
}

function runPlanSearch() {
  var base = document.getElementById('plan-s-base').value;
  if (!base) { _planToast('Selecciona una base primero', 'err'); return; }
  var sigla = _pEvtActual ? (_pEvtActual.sigla || '') : '';
  var res = document.getElementById('plan-s-results');
  var lbl = document.getElementById('plan-s-base').options[document.getElementById('plan-s-base').selectedIndex].text;

  // V4-FIX: Reset state on base change
  _searchAllData = [];
  window._pSearchRes = [];
  document.getElementById('plan-s-texto').value = '';
  var summary = document.getElementById('plan-s-summary');
  if (summary) { summary.style.display = 'none'; summary.innerHTML = ''; }
  document.getElementById('plan-s-info').textContent = '';
  _searchUrgFilter = '';
  _updateUrgFilters();

  res.innerHTML = '<div style="display:flex;align-items:center;justify-content:center;padding:50px;gap:10px;"><div class="plan-spin"></div><span style="font-size:12px;color:#94a3b8;">Buscando en ' + lbl + '...</span></div>';

  google.script.run
    .withSuccessHandler(function(r) {
      if (!r || !r.success) {
        res.innerHTML = '<div style="text-align:center;padding:30px;"><p style="color:#dc2626;font-size:12px;">' + (r ? r.error : 'Error') + '</p></div>';
        return;
      }
      _searchAllData = (r.empleados || []).map(function(e) {
        var dias = e.diasVencido;
        var st = String(e.statusCurso || '').toUpperCase();
        if (dias !== undefined && dias !== '' && Number(dias) <= 0) e.urgencia = 'VENCIDO';
        else if (dias !== undefined && dias !== '' && Number(dias) <= 90) e.urgencia = 'POR_VENCER';
        else if (st.indexOf('VENCID') >= 0) e.urgencia = 'VENCIDO';
        else if (st.indexOf('HABILIT') >= 0 || st.indexOf('VIGENT') >= 0) e.urgencia = 'HABILITADO';
        else e.urgencia = 'SIN_DATO';
        return e;
      });
      window._pSearchRes = _searchAllData;
      // V4-FIX: Show summary counts like openPlanSearch does
      var venc = 0, porv = 0, hab = 0, sd = 0;
      for (var i = 0; i < _searchAllData.length; i++) {
        var u = _searchAllData[i].urgencia;
        if (u === 'VENCIDO') venc++;
        else if (u === 'POR_VENCER') porv++;
        else if (u === 'HABILITADO') hab++;
        else sd++;
      }
      var sh = '<div style="display:flex;gap:10px;flex-wrap:wrap;align-items:center;">';
      sh += '<span style="font-size:11px;font-weight:700;color:#1e293b;">' + _searchAllData.length + ' candidatos</span>';
      if (venc > 0) sh += '<span style="font-size:10px;font-weight:700;padding:2px 10px;border-radius:20px;background:#fef2f2;color:#dc2626;">' + venc + ' vencidos</span>';
      if (porv > 0) sh += '<span style="font-size:10px;font-weight:700;padding:2px 10px;border-radius:20px;background:#fffbeb;color:#d97706;">' + porv + ' por vencer</span>';
      if (hab > 0) sh += '<span style="font-size:10px;font-weight:700;padding:2px 10px;border-radius:20px;background:#f0fdf4;color:#16a34a;">' + hab + ' habilitados</span>';
      if (sd > 0) sh += '<span style="font-size:10px;font-weight:600;padding:2px 10px;border-radius:20px;background:#f8fafc;color:#94a3b8;">' + sd + ' sin dato</span>';
      sh += '</div>';
      summary.innerHTML = sh;
      summary.style.display = 'block';
      document.getElementById('plan-s-info').textContent = _searchAllData.length + ' resultados en ' + lbl + (sigla ? ' (curso: ' + sigla + ')' : '');
      _renderSearchResults(_searchAllData);
      _updateUrgFilters();
    })
    .withFailureHandler(function(e) {
      res.innerHTML = '<p style="color:#dc2626;text-align:center;padding:30px;">' + (e.message || e) + '</p>';
    })
    .buscarEmpleadosPlan(base, { texto: '', sigla: sigla, soloVencidos: false });
}

function _renderSearchResults(emps) {
  var res = document.getElementById('plan-s-results');
  if (!emps || !emps.length) {
    res.innerHTML = '<div style="text-align:center;padding:50px;"><i class="ph ph-user-minus" style="font-size:32px;color:#cbd5e1;display:block;margin-bottom:10px;"></i><p style="font-size:13px;color:#94a3b8;font-weight:600;">Sin candidatos</p></div>';
    return;
  }
  var mx = 200;
  var URGENCIA_COLORS = {
    'VENCIDO': { bg: '#fef2f2', border: '#fecaca', color: '#dc2626', label: 'Vencido', icon: 'ph-warning-circle', accent: '#dc2626' },
    'POR_VENCER': { bg: '#fffbeb', border: '#fde68a', color: '#d97706', label: 'Por vencer', icon: 'ph-clock', accent: '#d97706' },
    'HABILITADO': { bg: '#f0fdf4', border: '#bbf7d0', color: '#16a34a', label: 'Habilitado', icon: 'ph-check-circle', accent: '#16a34a' },
    'SIN_DATO': { bg: '#f8fafc', border: '#e2e8f0', color: '#94a3b8', label: 'Sin dato', icon: 'ph-question', accent: '#94a3b8' }
  };

  // Mapa de BPs ya asignados al evento actual
  var _bpsAsignados = {};
  if (window._prRegs) {
    for (var a = 0; a < window._prRegs.length; a++) {
      var bp = String(window._prRegs[a].bp || '').trim();
      if (bp) _bpsAsignados[bp] = window._prRegs[a].estadoContacto || 'SELECCIONADO';
    }
  }

  var h = '<div style="display:flex;flex-direction:column;gap:6px;padding:4px 0;">';
  for (var i = 0; i < Math.min(emps.length, mx); i++) {
    var e = emps[i];
    var urg = URGENCIA_COLORS[e.urgencia] || URGENCIA_COLORS['SIN_DATO'];
    var yaAsignado = _bpsAsignados[String(e.bp || '').trim()];

    // Dias text
    var diasTxt = '';
    if (e.diasVencido !== undefined && e.diasVencido !== '') {
      var dv = Number(e.diasVencido);
      diasTxt = dv <= 0 ? Math.abs(dv) + 'd atrasado' : dv + 'd restantes';
    }

    // Fecha vencimiento
    var vencTxt = '';
    if (e.vencimiento) {
      var vStr = String(e.vencimiento).trim();
      try {
        // Caso 1: Formato ISO o YYYY-MM-DD (ej: "2026-06-02T05:00:00Z" o "2026-06-02")
        var matchISO = vStr.match(/^(\d{4})-(\d{2})-(\d{2})/);
        // Caso 2: Formato DD/MM/YYYY o D/M/YYYY (ej: "02/06/2026" o "2/6/2026")
        var matchDMY = vStr.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})/);

        if (matchISO) {
          vencTxt = matchISO[3] + '/' + matchISO[2] + '/' + matchISO[1];
        } else if (matchDMY) {
          // Si ya viene formateado, solo forzamos que tenga 2 dígitos (ceros a la izquierda)
          vencTxt = String(matchDMY[1]).padStart(2, '0') + '/' + 
                    String(matchDMY[2]).padStart(2, '0') + '/' + 
                    matchDMY[3];
        } else {
          // Fallback final: Dejar que Date intente parsear strings verbales (ej: "Tue Jun 02 2026...")
          var vDate = new Date(vStr);
          if (!isNaN(vDate.getTime())) {
            vencTxt = String(vDate.getDate()).padStart(2, '0') + '/' +
                      String(vDate.getMonth() + 1).padStart(2, '0') + '/' +
                      vDate.getFullYear();
          } else {
            vencTxt = vStr;
          }
        }
      } catch(ex) {
        vencTxt = vStr;
      }
    }

    // Card styles
    var cardBg = yaAsignado ? '#f0fdf4' : '#ffffff';
    var cardBorder = yaAsignado ? '#86efac' : urg.border;
    var cardOpacity = yaAsignado ? 'opacity:0.7;' : '';
    var clickAction = yaAsignado ? '' : 'onclick="_togChk(' + i + ')"';

    h += '<div ' + clickAction + ' style="display:flex;align-items:center;gap:12px;padding:10px 14px;background:' + cardBg + ';border:1px solid ' + cardBorder + ';border-left:4px solid ' + urg.accent + ';border-radius:10px;cursor:' + (yaAsignado ? 'default' : 'pointer') + ';transition:all .15s;' + cardOpacity + '" data-urgencia="' + e.urgencia + '" data-testid="plan-search-row-' + i + '"';
    if (!yaAsignado) h += ' onmouseover="this.style.boxShadow=\'0 2px 8px rgba(0,0,0,0.08)\';this.style.transform=\'translateY(-1px)\'" onmouseout="this.style.boxShadow=\'none\';this.style.transform=\'none\'"';
    h += '>';

    // Checkbox
    if (yaAsignado) {
      h += '<input type="checkbox" disabled style="opacity:0.3;flex-shrink:0;width:16px;height:16px;" data-testid="plan-emp-chk-' + i + '" />';
    } else {
      h += '<input type="checkbox" id="pchk-' + i + '" onclick="event.stopPropagation();_updSel()" style="flex-shrink:0;width:16px;height:16px;accent-color:' + urg.accent + ';" data-testid="plan-emp-chk-' + i + '" />';
    }

    // Main content block
    h += '<div style="flex:1;min-width:0;display:flex;flex-wrap:wrap;gap:4px 16px;align-items:center;">';

    // Name + BP row
    h += '<div style="min-width:200px;flex:2;">';
    h += '<div style="font-size:12px;font-weight:700;color:#1e293b;line-height:1.3;">' + e.nombre + '</div>';
    h += '<div style="font-size:10px;color:#94a3b8;font-weight:600;margin-top:1px;">BP ' + e.bp;
    if (e.pais) h += ' &middot; ' + e.pais;
    h += '</div>';
    h += '</div>';

    // Area + Jefe
    h += '<div style="min-width:140px;flex:1;">';
    if (e.area) h += '<div style="font-size:10px;color:#64748b;"><i class="ph ph-buildings" style="font-size:10px;color:#94a3b8;margin-right:2px;"></i>' + e.area + '</div>';
    if (e.jefe) h += '<div style="font-size:10px;color:#64748b;margin-top:1px;"><i class="ph ph-user" style="font-size:10px;color:#94a3b8;margin-right:2px;"></i>' + e.jefe + '</div>';
    h += '</div>';

    // Vencimiento + Dias
    h += '<div style="min-width:110px;text-align:right;">';
    if (vencTxt) {
      h += '<div style="font-size:11px;font-weight:700;color:' + urg.color + ';">' + vencTxt + '</div>';
    }
    if (diasTxt) {
      h += '<div style="font-size:9px;font-weight:600;color:' + urg.color + ';opacity:0.75;margin-top:1px;">' + diasTxt + '</div>';
    }
    if (!vencTxt && !diasTxt) {
      h += '<div style="font-size:10px;color:#cbd5e1;">—</div>';
    }
    h += '</div>';

    // Urgencia badge
    h += '<div style="min-width:90px;text-align:center;">';
    h += '<span style="display:inline-flex;align-items:center;gap:3px;font-size:10px;font-weight:700;padding:3px 10px;border-radius:20px;background:' + urg.bg + ';color:' + urg.color + ';border:1px solid ' + urg.border + ';"><i class="ph ' + urg.icon + '" style="font-size:11px;"></i> ' + urg.label + '</span>';
    h += '</div>';

    // Ya asignado badge (if applicable)
    if (yaAsignado) {
      var stLabel = (ESTADOS_V2[yaAsignado] || {}).label || yaAsignado;
      h += '<div style="min-width:80px;text-align:center;">';
      h += '<span style="font-size:9px;font-weight:700;padding:3px 10px;border-radius:20px;background:#dcfce7;color:#166534;border:1px solid #86efac;"><i class="ph ph-check-circle" style="font-size:10px;"></i> ' + stLabel + '</span>';
      h += '</div>';
    }

    // V2.8: Plan status from suggestions (REPROBADO/NO_ASISTIO)
    if (e._planStatus && !yaAsignado) {
      var psInfo = { 'REPROBADO': { label: 'Reprobo (repetir)', color: '#dc2626', bg: '#fee2e2' }, 'NO_ASISTIO': { label: 'No asistio prev.', color: '#ea580c', bg: '#fff7ed' }, 'PROGRAMADO': { label: 'Ya programado', color: '#8b5cf6', bg: '#ede9fe' } };
      var ps = psInfo[e._planStatus] || {};
      if (ps.label) {

function _applyLocalFilters() {
  var txt = (document.getElementById('plan-s-texto').value || '').toUpperCase().trim();
  var filtered = [];
  for (var i = 0; i < _searchAllData.length; i++) {
    var e = _searchAllData[i];
    if (_searchUrgFilter && e.urgencia !== _searchUrgFilter) continue;
    if (txt) {
      var hayMatch = (e.nombre || '').toUpperCase().indexOf(txt) >= 0 ||
                     (e.bp || '').indexOf(txt) >= 0 ||
                     (e.area || '').toUpperCase().indexOf(txt) >= 0 ||
                     (e.pais || '').toUpperCase().indexOf(txt) >= 0;
      if (!hayMatch) continue;
    }
    filtered.push(e);
  }
  window._pSearchRes = filtered;
  _renderSearchResults(filtered);
  document.getElementById('plan-s-info').textContent = filtered.length + ' de ' + _searchAllData.length + ' candidatos' + (_searchUrgFilter ? ' (' + _searchUrgFilter.replace('_', ' ').toLowerCase() + ')' : '') + (txt ? ' — filtro: "' + txt.toLowerCase() + '"' : '');
}

function _updateUrgFilters() {
  var counts = { 'VENCIDO': 0, 'POR_VENCER': 0, '': 0 };
  for (var i = 0; i < _searchAllData.length; i++) {
    var u = _searchAllData[i].urgencia;
    if (u === 'VENCIDO') counts['VENCIDO']++;
    else if (u === 'POR_VENCER') counts['POR_VENCER']++;
    counts['']++;
  }
  var fVenc = document.getElementById('plan-s-f-venc');
  var fPorv = document.getElementById('plan-s-f-porv');
  var fAll = document.getElementById('plan-s-f-all');
  if (fVenc) { fVenc.textContent = 'Vencidos (' + counts['VENCIDO'] + ')'; fVenc.style.background = _searchUrgFilter === 'VENCIDO' ? '#dc2626' : ''; fVenc.style.color = _searchUrgFilter === 'VENCIDO' ? 'white' : ''; }
  if (fPorv) { fPorv.textContent = 'Por vencer (' + counts['POR_VENCER'] + ')'; fPorv.style.background = _searchUrgFilter === 'POR_VENCER' ? '#d97706' : ''; fPorv.style.color = _searchUrgFilter === 'POR_VENCER' ? 'white' : ''; }
  if (fAll) { fAll.textContent = 'Todos (' + counts[''] + ')'; fAll.style.background = !_searchUrgFilter ? 'var(--pl-blue)' : ''; fAll.style.color = !_searchUrgFilter ? 'white' : ''; }
}

    var c = document.getElementById('plan-wl-list');
    if (!list || !list.length) { c.innerHTML = '<p style="font-size:10px;color:#94a3b8;">Sin correos</p>'; return; }
    var h = '';
    list.forEach(function(e, i) {
      h += '<div style="display:flex;justify-content:space-between;align-items:center;padding:4px 8px;background:' + (i % 2 === 0 ? '#f8fafc' : '#fff') + ';border-radius:4px;">';
      h += '<span style="font-size:11px;">' + e + '</span>';
      h += '<button onclick="rmWlPlan(\'' + e + '\')" style="background:none;border:none;color:#dc2626;cursor:pointer;opacity:.35;font-size:11px;padding:2px;" onmouseover="this.style.opacity=1" onmouseout="this.style.opacity=.35"><i class="ph ph-x"></i></button></div>';
    });
    c.innerHTML = h;
    window._pWlCache = list;
  }).getPlanningWhitelist();
}
function addWlPlan() {
  var inp = document.getElementById('plan-wl-inp'), email = (inp.value || '').trim().toLowerCase();
  if (!email || email.indexOf('@') < 0) { _planToast('Correo invalido', 'err'); return; }
  var list = (window._pWlCache || []).slice();
  if (list.indexOf(email) >= 0) { _planToast('Ya existe', 'err'); return; }
  list.push(email); inp.value = '';
  google.script.run.withSuccessHandler(function() { loadWlPlan(); _planToast('Correo agregado', 'ok'); }).setPlanningWhitelist(list);
}
function rmWlPlan(email) {
  var list = (window._pWlCache || []).filter(function(e) { return e !== email; });
  google.script.run.withSuccessHandler(function() { loadWlPlan(); _planToast('Eliminado', 'ok'); }).setPlanningWhitelist(list);
}

function _initTestModePlan() {
  google.script.run
    .withSuccessHandler(function(cfg) {
      _renderTestModeUI(cfg);
    })
    .getTestModeConfig();
}
function _renderTestModeUI(cfg) {
  if (!cfg) return;
  var sel = document.getElementById('plan-test-mode-select');
  var inp = document.getElementById('plan-test-email-input');
  var info = document.getElementById('plan-test-mode-info');
  var banner = document.getElementById('plan-test-banner');
  if (sel) sel.value = cfg.modo || 'OFF';
  if (inp) inp.value = cfg.emailTest || '';
  _previewTestMode();
  // Banner
  if (banner) {
    if (cfg.activo) {
      if (cfg.modo === 'SIMULACION') {
        banner.style.display = 'block';
        banner.style.background = 'linear-gradient(135deg,#fef3c7,#fde68a)';
        banner.style.border = '2px solid #f59e0b';
        banner.style.color = '#92400e';
        banner.innerHTML = '<i class="ph ph-flask" style="margin-right:6px;"></i>MODO SIMULACION — Los correos NO se envian. El flujo se ejecuta normalmente.';
      } else if (cfg.modo === 'TEST') {
        banner.style.display = 'block';
        banner.style.background = 'linear-gradient(135deg,#fef2f2,#fecaca)';
        banner.style.border = '2px solid #f87171';
        banner.style.color = '#991b1b';
        banner.innerHTML = '<i class="ph ph-flask" style="margin-right:6px;"></i>MODO TEST — Correos redirigidos a: <strong>' + (cfg.emailTest || '(sin configurar)') + '</strong>';
      }
    } else {
      banner.style.display = 'none';
    }
  }
}
function _previewTestMode() {
  var sel = document.getElementById('plan-test-mode-select');
  var info = document.getElementById('plan-test-mode-info');
  var inp = document.getElementById('plan-test-email-input');
  if (!sel || !info) return;
  var modo = sel.value;
  if (modo === 'OFF') {
    info.innerHTML = '<i class="ph ph-check-circle" style="color:#16a34a;"></i> Produccion: correos se envian a destinatarios reales.';
    if (inp) inp.style.display = 'none';
  } else if (modo === 'SIMULACION') {
    info.innerHTML = '<i class="ph ph-pause-circle" style="color:#d97706;"></i> Simulacion: <strong>NO se envian correos</strong>. El flujo se ejecuta normalmente. Registros marcados como TEST.';
    if (inp) inp.style.display = 'none';
  } else if (modo === 'TEST') {
    info.innerHTML = '<i class="ph ph-arrow-bend-double-up-right" style="color:#dc2626;"></i> Test: correos van <strong>SOLO al email de prueba</strong>. Registros marcados como TEST.';
    if (inp) inp.style.display = 'block';
  }
}
function guardarTestModePlan() {
  var sel = document.getElementById('plan-test-mode-select');
  var inp = document.getElementById('plan-test-email-input');
  var modo = sel ? sel.value : 'OFF';
  var email = inp ? inp.value.trim() : '';
  if (modo === 'TEST' && (!email || email.indexOf('@') < 0)) {
    _planToast('Ingresa un email valido para modo TEST', 'err');
    return;
  }
  google.script.run
    .withSuccessHandler(function(r) {
      if (r && r.success) {
        _renderTestModeUI(r.config);
        _planToast('Modo de prueba: ' + modo, 'ok');
      } else {
        _planToast('Error: ' + (r ? r.error : 'sin respuesta'), 'err');
      }
    })
    .withFailureHandler(function(e) { _planToast('Error: ' + (e.message || e), 'err'); })
    .setTestModeConfig({ modo: modo, emailTest: email });
}
function limpiarDatosTestPlan() {
  _planConfirmShow('Esto eliminara TODOS los registros marcados como TEST en PLAN_REGISTRO. Continuar?', function() {
  google.script.run
    .withSuccessHandler(function(r) {

        _planToast(r.eliminados + ' registros TEST eliminados', 'ok');

        _planToast('Error: ' + (r ? r.error : ''), 'err');
      }
    })
    .withFailureHandler(function(e) { _planToast('Error: ' + (e.message || e), 'err'); })
    .limpiarRegistrosTest();
  });
}

    var body = document.getElementById('plan-kpi-modal-body');

    }

    // Group by status
    var statusCounts = {};

      var sc = stColors[st] || '#64748b';
      h += '<span style="background:' + sc + '18;color:' + sc + ';padding:4px 10px;border-radius:20px;font-size:11px;font-weight:700;">' + st.replace(/_/g, ' ') + ': ' + statusCounts[st] + '</span>';

function _planKpiRegsTable(regs) {
  if (!regs.length) return '<p style="text-align:center;color:#94a3b8;font-size:12px;">Sin resultados</p>';
  var stColors = { 'SELECCIONADO':'#64748b', 'PRECONFIRMADO_JEFE':'#8b5cf6', 'CITADO':'#3b82f6', 'CONFIRMADO':'#22c55e', 'NO_CONTACTADO':'#94a3b8', 'RECHAZADO':'#ef4444' };
  var h = '<table style="width:100%;border-collapse:collapse;font-size:11px;">';
  h += '<thead><tr style="background:#f8fafc;"><th style="text-align:left;padding:8px;font-weight:700;color:#475569;">Nombre</th><th style="padding:8px;">BP</th><th style="padding:8px;">Sigla</th><th style="padding:8px;">Fecha</th><th style="padding:8px;">Estado</th><th style="padding:8px;">Estado Operativo</th></tr></thead><tbody>';
  for (var i = 0; i < regs.length; i++) {
    var r = regs[i];
    var sc = stColors[r.estadoContacto] || '#64748b';
    var estOp = r.estadoOperativo || 'PRE_CURSO';
    var opI = ESTADOS_OP[estOp] || {};
    h += '<tr style="border-bottom:1px solid #f1f5f9;">';
    h += '<td style="padding:7px 8px;font-weight:600;color:#1e293b;">' + (r.nombre || '') + '</td>';
    h += '<td style="padding:7px 8px;text-align:center;color:#64748b;">' + (r.bp || '') + '</td>';
    h += '<td style="padding:7px 8px;text-align:center;font-weight:700;">' + (r.sigla || '') + '</td>';
    h += '<td style="padding:7px 8px;text-align:center;color:#64748b;">' + _fmtFechaCorta(r.fechaClase) + '</td>';
    h += '<td style="padding:7px 8px;text-align:center;"><span style="background:' + sc + '18;color:' + sc + ';padding:2px 8px;border-radius:12px;font-size:10px;font-weight:700;">' + (r.estadoContacto || '').replace(/_/g, ' ') + '</span></td>';
    // V2.8: Estado operativo
    h += '<td style="padding:7px 8px;text-align:center;">';
    if (estOp !== 'PRE_CURSO' && opI.label) {
      h += '<span style="display:inline-flex;align-items:center;gap:2px;background:' + (opI.bg || '#f1f5f9') + ';color:' + (opI.color || '#64748b') + ';padding:2px 8px;border-radius:12px;font-size:9px;font-weight:700;">';
      if (opI.icon) h += '<i class="ph ' + opI.icon + '" style="font-size:9px;"></i>';
      h += opI.label + '</span>';
    } else {
      h += '<span style="color:#cbd5e1;font-size:10px;">Pre-curso</span>';
    }
    h += '</td>';
    h += '</tr>';
  }
  h += '</tbody></table>';
  return h;
}

function _planKpiFilterRegs(val) {
  var regs = window._planKpiRegsData || [];
  if (!val) { document.getElementById('plan-kpi-regs-body').innerHTML = _planKpiRegsTable(regs); return; }
  var q = val.toLowerCase();
  var filtered = regs.filter(function(r) {
    return (r.nombre || '').toLowerCase().indexOf(q) >= 0 || (r.bp || '').toLowerCase().indexOf(q) >= 0 || (r.sigla || '').toLowerCase().indexOf(q) >= 0;
  });
  document.getElementById('plan-kpi-regs-body').innerHTML = _planKpiRegsTable(filtered);
}

function _renderGestion() {
  var board = document.getElementById('gst-board');
  var fCurso = document.getElementById('gst-f-curso').value;

  var fTexto = (document.getElementById('gst-f-texto').value || '').toUpperCase().trim();

  // Filter
  var filtered = [];
  for (var i = 0; i < _gstData.length; i++) {
    var r = _gstData[i];
    if (fCurso && r.sigla !== fCurso) continue;
    if (fPais && r.pais !== fPais) continue;
    if (fTexto && (r.nombre || '').toUpperCase().indexOf(fTexto) < 0 && (r.bp || '').indexOf(fTexto) < 0) continue;
    filtered.push(r);
  }

  // Group by kanban columns
  var cols = [
    { key: 'SELECCIONADO', label: 'Seleccionados', color: '#f59e0b', bg: '#fffbeb', border: '#fde68a', icon: 'ph-user-list', items: [] },
    { key: 'PRE_CONFIRMADO', label: 'Pre-confirmados', color: '#8b5cf6', bg: '#faf5ff', border: '#ddd6fe', icon: 'ph-user-check', items: [] },
    { key: 'CONFIRMADO', label: 'Confirmados', color: '#22c55e', bg: '#f0fdf4', border: '#bbf7d0', icon: 'ph-check-circle', items: [] },
    { key: 'RECHAZADO', label: 'Rechazados', color: '#ef4444', bg: '#fef2f2', border: '#fecaca', icon: 'ph-x-circle', items: [] },
    { key: 'POST_CURSO', label: 'Post-Curso', color: '#0284c7', bg: '#f0f9ff', border: '#bae6fd', icon: 'ph-graduation-cap', items: [] }
  ];
  var colMap = {};
  for (var c = 0; c < cols.length; c++) colMap[cols[c].key] = c;
  for (var j = 0; j < filtered.length; j++) {
    var idx = colMap[filtered[j].grupo];
    if (idx !== undefined) cols[idx].items.push(filtered[j]);
  }

  var h = '';
  for (var k = 0; k < cols.length; k++) {
    var col = cols[k];
    h += '<div style="background:' + col.bg + ';border:1px solid ' + col.border + ';border-radius:12px;min-height:200px;display:flex;flex-direction:column;" data-testid="gst-col-' + col.key + '">';
    // Column header
    h += '<div style="padding:10px 14px;border-bottom:1px solid ' + col.border + ';display:flex;align-items:center;justify-content:space-between;position:sticky;top:0;background:' + col.bg + ';border-radius:12px 12px 0 0;z-index:1;">';
    h += '<div style="display:flex;align-items:center;gap:6px;"><i class="ph ' + col.icon + '" style="color:' + col.color + ';font-size:15px;"></i>';
    h += '<span style="font-size:12px;font-weight:800;color:' + col.color + ';">' + col.label + '</span></div>';
    h += '<span style="font-size:11px;font-weight:900;color:' + col.color + ';background:white;padding:2px 8px;border-radius:20px;">' + col.items.length + '</span>';
    h += '</div>';
    // Cards
    h += '<div style="padding:8px;flex:1;overflow-y:auto;max-height:500px;display:flex;flex-direction:column;gap:6px;">';
    if (!col.items.length) {
      h += '<div style="text-align:center;padding:20px;color:' + col.color + ';font-size:11px;opacity:.5;">Sin registros</div>';

      var card = col.items[m];
      h += _buildGstCard(card, col.color);
    }
    h += '</div></div>';
  }

}

function _getFilteredPreconf() {
  var fSigla = document.getElementById('preconf-f-sigla').value;
  var fPais = document.getElementById('preconf-f-pais').value;
  var fJefe = document.getElementById('preconf-f-jefe').value;
  var fTexto = (document.getElementById('preconf-f-texto').value || '').toUpperCase().trim();
  return _preconfData.filter(function(r) {
    if (fSigla && r.sigla !== fSigla) return false;
    if (fPais && r.pais !== fPais) return false;
    if (fJefe && r.jefe !== fJefe) return false;
    if (fTexto) {
      var hay = (r.nombre || '').toUpperCase().indexOf(fTexto) >= 0 || (r.bp || '').toUpperCase().indexOf(fTexto) >= 0;
      if (!hay) return false;
    }
    return true;
  });
}

function _filterRegistros() {
  var fAccion = document.getElementById('reg-f-accion').value;
  var fTexto = (document.getElementById('reg-f-texto').value || '').toUpperCase().trim();
  var filtered = [];

  for (var i = 0; i < _regData.length; i++) {
    var r = _regData[i];
    // Prefilter from Kanban
    if (_regPrefilter) {
      if (_regPrefilter.eventoId && r.eventoId !== _regPrefilter.eventoId) continue;
      if (_regPrefilter.bp && r.detalle.indexOf(_regPrefilter.bp) < 0 && !_isEventLevelAction(r.accion)) continue;
    }
    if (fAccion && r.accion !== fAccion) continue;
    if (fTexto) {
      var hayMatch = r.usuario.toUpperCase().indexOf(fTexto) >= 0 ||
                     r.detalle.toUpperCase().indexOf(fTexto) >= 0 ||
                     r.accionLabel.toUpperCase().indexOf(fTexto) >= 0 ||
                     (r.eventoId || '').indexOf(fTexto) >= 0;
      if (!hayMatch) continue;
    }
    filtered.push(r);
  }

  // Stats
  var statsEl = document.getElementById('reg-stats');
  statsEl.innerHTML = '<div style="font-size:10px;color:#94a3b8;font-weight:600;">' + filtered.length + ' de ' + _regData.length + ' registros' + (fAccion || fTexto || _regPrefilter ? ' (filtrado)' : '') + '</div>';

  _renderRegistrosTable(filtered);
}

function _isEventLevelAction(accion) {
  return accion === 'REGISTRAR_SELECCION' || accion === 'APROBACION_JEFE' ||
         accion === 'NOTIFICAR_JEFE' || accion === 'CREAR_LIBRO' ||
         accion.indexOf('ENVIO_CORREO') === 0 || accion === 'CAMBIO_ESTADO_BULK';
}

function _segPlanPoblarFiltros(registros) {
  var cursos = {}, paises = {}, jefes = {}, areas = {};
  for (var i = 0; i < registros.length; i++) {
    var r = registros[i];
    if (r.sigla) cursos[r.sigla] = true;
    if (r.pais) paises[r.pais] = true;
    if (r.jefe) jefes[r.jefe] = true;
    if (r.area) areas[r.area] = true;
  }
  _segPlanLlenarSel('segplan-f-curso', cursos, 'Curso: Todos');
  _segPlanLlenarSel('segplan-f-pais', paises, 'Pais: Todos');
  _segPlanLlenarSel('segplan-f-jefe', jefes, 'Jefe: Todos');
  _segPlanLlenarSel('segplan-f-area', areas, 'Area: Todas');
}

function _segPlanLlenarSel(id, valuesObj, defaultLabel) {

  var keys = Object.keys(valuesObj).sort();

    sel.innerHTML += '<option value="' + keys[i] + '"' + (keys[i] === cur ? ' selected' : '') + '>' + keys[i] + '</option>';
  }
}

function _segPlanFiltrar() {
  var fEstado = (document.getElementById('segplan-f-estado') || {}).value || '';
  var fCurso = (document.getElementById('segplan-f-curso') || {}).value || '';
  var fPais = (document.getElementById('segplan-f-pais') || {}).value || '';
  var fJefe = (document.getElementById('segplan-f-jefe') || {}).value || '';
  var fArea = (document.getElementById('segplan-f-area') || {}).value || '';
  var fTexto = ((document.getElementById('segplan-f-texto') || {}).value || '').toLowerCase().trim();

  var filtrados = [];
  for (var i = 0; i < _segPlanData.length; i++) {
    var r = _segPlanData[i];
    if (fEstado && r.estado !== fEstado) continue;
    if (fCurso && r.sigla !== fCurso) continue;
    if (fPais && r.pais !== fPais) continue;
    if (fJefe && r.jefe !== fJefe) continue;
    if (fArea && r.area !== fArea) continue;
    if (fTexto) {
      var hayMatch = (r.nombre || '').toLowerCase().indexOf(fTexto) >= 0 ||
        (r.bp || '').toLowerCase().indexOf(fTexto) >= 0 ||
        (r.email || '').toLowerCase().indexOf(fTexto) >= 0 ||
        (r.sigla || '').toLowerCase().indexOf(fTexto) >= 0;
      if (!hayMatch) continue;
    }
    filtrados.push(r);
  }
  _segPlanRenderTabla(filtrados);
  var countEl = document.getElementById('segplan-count');
  if (countEl) countEl.textContent = filtrados.length + ' de ' + _segPlanData.length + ' registros' + (fEstado || fCurso || fPais || fJefe || fArea || fTexto ? ' (filtrado)' : '');

function _baseReload() {

  tbl.innerHTML = '<div style="text-align:center;padding:50px;"><div class="plan-spin" style="margin:0 auto;"></div><p style="font-size:11px;color:#94a3b8;margin-top:8px;">Cargando base de colaboradores...</p></div>';

      tbl.innerHTML = '<div style="text-align:center;padding:30px;color:#dc2626;font-size:12px;">' + (r ? r.error : 'Error') + '</div>';
      return;
    }
    _baseData = r.colaboradores || [];
    _baseSiglasEnPlan = r.siglasEnPlan || [];
    _baseLoaded = true;
    _baseRenderKpis(r.kpis || {});
    _basePoblarFiltros(_baseData);

    _baseLastUpdateTime = new Date();
    _baseUpdateTicker();

  }).withFailureHandler(function(e) {

  }).getBaseColaboradores();

// V7: Auto-refresh
function _baseStartAutoRefresh() {
  _baseStopAutoRefresh();
  _baseAutoRefreshTimer = setInterval(function() {
    console.log('[BASE-V7] Auto-refresh...');
    _baseReload();
  }, _BASE_REFRESH_MS);
  // Ticker para "hace X min"
  _baseLastUpdateTicker = setInterval(_baseUpdateTicker, 30000);
}

function _baseStopAutoRefresh() {
  if (_baseAutoRefreshTimer) { clearInterval(_baseAutoRefreshTimer); _baseAutoRefreshTimer = null; }
  if (_baseLastUpdateTicker) { clearInterval(_baseLastUpdateTicker); _baseLastUpdateTicker = null; }
}

function _baseUpdateTicker() {
  var el = document.getElementById('base-lastupdate-text');
  if (!el || !_baseLastUpdateTime) return;
  var diff = Math.floor((new Date() - _baseLastUpdateTime) / 1000);
  var txt = 'Ahora';
  if (diff >= 60) txt = 'Hace ' + Math.floor(diff / 60) + ' min';
  if (diff >= 3600) txt = 'Hace ' + Math.floor(diff / 3600) + 'h';
  el.textContent = txt + ' · Auto-refresh 3min';
}

var _ssBaseCurso = null, _ssBaseArea = null, _ssBaseJefe = null;

function _basePoblarFiltros(data) {
  var areas = {}, jefes = {};
  for (var i = 0; i < data.length; i++) {
    if (data[i].area) areas[data[i].area] = true;
    if (data[i].jefe) jefes[data[i].jefe] = true;
  }
  // Cursos = siglas en plan + cursos base (forecast)
  var cursoObj = {};
  for (var s = 0; s < _baseSiglasEnPlan.length; s++) cursoObj[_baseSiglasEnPlan[s]] = true;
  for (var j = 0; j < data.length; j++) {
    if (data[j].cursoBase) cursoObj[data[j].cursoBase] = true;
  }

  var cursoArr = Object.keys(cursoObj).sort();
  var areaArr = Object.keys(areas).sort();
  var jefeArr = Object.keys(jefes).sort();

  if (!_ssBaseCurso) {
    _ssBaseCurso = _ssCreate('base-ss-curso', cursoArr, 'Curso...', function() { _baseFiltrar(); });
  } else {
    _ssBaseCurso.setOptions(cursoArr);
  }
  if (!_ssBaseArea) {
    _ssBaseArea = _ssCreate('base-ss-area', areaArr, 'Area...', function() { _baseFiltrar(); });
  } else {
    _ssBaseArea.setOptions(areaArr);
  }
  if (!_ssBaseJefe) {
    _ssBaseJefe = _ssCreate('base-ss-jefe', jefeArr, 'Jefe...', function() { _baseFiltrar(); });
  } else {
    _ssBaseJefe.setOptions(jefeArr);
  }
}

function _baseFiltrar() {
  var fTexto = ((document.getElementById('base-f-texto') || {}).value || '').toLowerCase().trim();
  var fPlan = (document.getElementById('base-f-planestado') || {}).value || '';
  var fArea = _ssBaseArea ? _ssBaseArea.getValue() : '';
  var fJefe = _ssBaseJefe ? _ssBaseJefe.getValue() : '';
  var fCurso = _ssBaseCurso ? _ssBaseCurso.getValue() : '';
  var fOrden = (document.getElementById('base-f-orden') || {}).value || 'nombre';

  var filtrados = [];
  for (var i = 0; i < _baseData.length; i++) {
    var r = _baseData[i];
    if (fArea && r.area !== fArea) continue;
    if (fJefe && r.jefe !== fJefe) continue;
    if (fPlan === 'EN_PLAN' && !r.planEstado) continue;
    if (fPlan === 'SIN_PLAN' && r.planEstado) continue;
    if (fPlan === 'CONFIRMADO' && r.planEstado !== 'CONFIRMADO') continue;
    if (fPlan === 'SELECCIONADO' && r.planEstado !== 'SELECCIONADO') continue;
    if (fPlan === 'RECHAZADO' && r.planEstado !== 'RECHAZADO') continue;
    // V7: Filtro por curso — busca en TODOS los cursos del empleado + plan
    if (fCurso) {
      var cursoMatch = false;
      if (r.planSigla === fCurso || r.cursoBase === fCurso) cursoMatch = true;
      if (!cursoMatch && r.cursos) {
        for (var cc = 0; cc < r.cursos.length; cc++) {
          if (r.cursos[cc].sigla === fCurso) { cursoMatch = true; break; }
        }
      }
      if (!cursoMatch && r.planCursos) {
        for (var pc = 0; pc < r.planCursos.length; pc++) {
          if (r.planCursos[pc].sigla === fCurso) { cursoMatch = true; break; }
        }
      }
      if (!cursoMatch) continue;
    }
    if (fTexto) {
      var textMatch = (r.nombre || '').toLowerCase().indexOf(fTexto) >= 0
        || (r.bp || '').indexOf(fTexto) >= 0
        || (r.planSigla || '').toLowerCase().indexOf(fTexto) >= 0
        || (r.cursoBase || '').toLowerCase().indexOf(fTexto) >= 0
        || (r.correo || '').toLowerCase().indexOf(fTexto) >= 0;
      if (!textMatch && r.cursos) {
        for (var tc = 0; tc < r.cursos.length; tc++) {
          if ((r.cursos[tc].sigla || '').toLowerCase().indexOf(fTexto) >= 0) { textMatch = true; break; }
        }
      }
      if (!textMatch) continue;
    }
    filtrados.push(r);
  }

  // V5: Ordenamiento
  filtrados.sort(function(a, b) {
    if (fOrden === 'estado') {
      var pa = a.planEstado ? 0 : 1, pb = b.planEstado ? 0 : 1;
      if (pa !== pb) return pa - pb;
      var ea = _estadoOrdenBase(a.planEstado), eb = _estadoOrdenBase(b.planEstado);
      if (ea !== eb) return ea - eb;
      return (a.nombre || '').localeCompare(b.nombre || '');
    }
    if (fOrden === 'curso') {
      var ca = (a.planSigla || a.cursoBase || 'ZZZ'), cb = (b.planSigla || b.cursoBase || 'ZZZ');
      if (ca !== cb) return ca.localeCompare(cb);
      return (a.nombre || '').localeCompare(b.nombre || '');
    }
    if (fOrden === 'urgencia') {
      var da = a.diasVencido !== undefined && a.diasVencido !== '' ? Number(a.diasVencido) : 9999;
      var db = b.diasVencido !== undefined && b.diasVencido !== '' ? Number(b.diasVencido) : 9999;
      return da - db;
    }
    return (a.nombre || '').localeCompare(b.nombre || '');
  });

  var activeFilters = (fTexto || fPlan || fArea || fJefe || fCurso) ? ' (filtrado)' : '';
  document.getElementById('base-count').textContent = filtrados.length + ' de ' + _baseData.length + ' colaboradores' + activeFilters;
  _baseRenderTabla(filtrados);
}

function _estadoOrdenBase(est) {
  var m = { 'CONFIRMADO': 1, 'PRE_CONFIRMADO': 2, 'SELECCIONADO': 3, 'PRECONFIRMADO_JEFE': 3, 'CITADO': 3, 'RECHAZADO': 4, 'CANCELADO': 5 };
  return m[est] || 99;
}

        h += '<div style="min-width:80px;text-align:center;"><span style="font-size:9px;font-weight:700;padding:2px 8px;border-radius:12px;background:' + ppc.bg + ';color:' + ppc.color + ';">' + (pp.estado || '') + '</span></div>';
        h += '<div style="min-width:95px;text-align:center;font-size:10px;color:#475569;">' + (pp.fechaClase || '—') + '</div>';
        h += '<div style="min-width:60px;text-align:right;font-size:9px;color:#cbd5e1;">—</div>';
        h += '</div>';
      }
    }

    h += '</div>'; // close cascada
    h += '</div>'; // close card
  }
  h += '</div>';
  if (regs.length > mx) h += '<p style="font-size:10px;color:#94a3b8;text-align:center;padding:8px;">Mostrando ' + mx + ' de ' + regs.length + '</p>';
  tbl.innerHTML = h;
}

// V113-FIX: Modal curso + refresh real + carga concurrente + Flatpickr 24h
function _abrirModalCurso(modo) {

  _cursoModo = modo;

  var modal = document.getElementById('planM-curso');
  var title = document.getElementById('planM-curso-title');

  if (!modal) {
    console.error('[PLAN] Modal planM-curso no existe');
    return;
  }

  if (title) {
    title.textContent = (modo === 'CREAR')
      ? 'Nueva Actividad'
      : 'Modificar Actividad';
  }

  // Mostrar modal
  modal.style.display = 'flex';

  // =========================================
  // V113-FIX: Flatpickr 24h seguro
  // =========================================
  if (typeof flatpickr !== 'undefined') {

    document.querySelectorAll(".flatpickr-time").forEach(function(el) {
      if (el._flatpickr) {
        el._flatpickr.destroy();
      }
    });

    flatpickr(".flatpickr-time", {
      enableTime: true,
      noCalendar: true,
      dateFormat: "H:i",
      time_24hr: true,
      allowInput: true
    });
  }

  // =========================================
  // V113-FIX: invalidar cache frontend
  // =========================================
  try {
    _eventsCache = {};
  } catch(e){}

  try {
    window._pSearchRes = [];
  } catch(e){}

  // =========================================
  // CARGA CONCURRENTE
  // =========================================
  var cursos = null;
  var instructores = null;
  var salas = null;

  function tryRender() {

    if (!cursos || !instructores || !salas) return;

    // =====================================
    // CURSOS
    // =====================================
    window._cursosBiblioteca = cursos || [];

    var selSigla = document.getElementById('curso-sigla');

    if (selSigla) {

      selSigla.innerHTML =
        '<option value="">Seleccionar curso</option>';

      cursos.forEach(function(c) {

        selSigla.innerHTML +=
          '<option value="' + c.sigla + '">' +
            c.sigla +
          '</option>';

      });
    }

    // =====================================
    // INSTRUCTORES
    // =====================================
    var selInst =
      document.getElementById('curso-instructor');

    if (selInst) {

      selInst.innerHTML =
        '<option value="">— Sin asignar —</option>';

      instructores.forEach(function(i) {

        selInst.innerHTML +=
          '<option value="' + i + '">' +
            i +
          '</option>';

      });
    }

    // =====================================
    // SALAS
    // =====================================
    var selSala =
      document.getElementById('curso-sala');

    if (selSala) {

      selSala.innerHTML =
        '<option value="">— Sin sala —</option>';

      salas.forEach(function(s) {

        selSala.innerHTML +=
          '<option value="' + s + '">' +
            s +
          '</option>';

      });
    }

    // =====================================
    // PREFILL MODIFICAR
    // =====================================
    if (modo === 'MODIFICAR' && _pEvtActual) {

      var ev = _pEvtActual || {};

      document.getElementById('curso-fecha').value =
        ev.fecha || '';

      document.getElementById('curso-tipo').value =
        (ev.tipo || 'CAPACITACION').toUpperCase();

      document.getElementById('curso-modalidad').value =
        (ev.modalidad || 'PRESENCIAL').toUpperCase();

      document.getElementById('curso-sigla').value =
        ev.sigla || '';

      document.getElementById('curso-titulo').value =
        ev.titulo || '';

      document.getElementById('curso-hora-inicio').value =
        ev.horaInicio || '';

      document.getElementById('curso-hora-fin').value =
        ev.horaFin || '';

      document.getElementById('curso-instructor').value =
        ev.instructor || '';

      document.getElementById('curso-participantes').value =
        ev.participantes || '';

      document.getElementById('curso-sala').value =
        ev.sala || '';

      document.getElementById('curso-dia').value =
        ev.dia || '';

      document.getElementById('curso-notas').value =
        ev.nota || '';

    } else {

      // =====================================
      // LIMPIAR CREAR
      // =====================================
      document.getElementById('curso-fecha').value = '';
      document.getElementById('curso-tipo').value = 'CAPACITACION';
      document.getElementById('curso-modalidad').value = 'PRESENCIAL';
      document.getElementById('curso-sigla').value = '';
      document.getElementById('curso-titulo').value = '';
      document.getElementById('curso-hora-inicio').value = '';
      document.getElementById('curso-hora-fin').value = '';
      document.getElementById('curso-instructor').value = '';
      document.getElementById('curso-participantes').value = '';
      document.getElementById('curso-sala').value = '';
      document.getElementById('curso-dia').value = '';
      document.getElementById('curso-notas').value = '';
    }
  }

  // =========================================
  // CURSOS
  // =========================================
  google.script.run
    .withSuccessHandler(function(r) {
      cursos = r || [];
      tryRender();
    })
    .withFailureHandler(function(e) {
      console.error('[PLAN] Error cursos', e);
      _planToast('Error cargando cursos', 'err');
    })
    .getCursosBiblioteca();

  // =========================================
  // INSTRUCTORES
  // =========================================
  google.script.run
    .withSuccessHandler(function(r) {
      instructores = r || [];
      tryRender();
    })
    .withFailureHandler(function(e) {
      console.error('[PLAN] Error instructores', e);
      _planToast('Error cargando instructores', 'err');
    })
    .getInstructores();

  // =========================================
  // SALAS
  // =========================================
  google.script.run
    .withSuccessHandler(function(r) {
      salas = r || [];
      tryRender();
    })
    .withFailureHandler(function(e) {
      console.error('[PLAN] Error salas', e);
      _planToast('Error cargando salas', 'err');
    })
    .getSalas();
}

function _onChangeSiglaCurso() {

  var sigla = document.getElementById('curso-sigla').value;

  if (!window._cursosBiblioteca) return;

  var curso = null;

  for (var i = 0; i < window._cursosBiblioteca.length; i++) {
    if (window._cursosBiblioteca[i].sigla === sigla) {
      curso = window._cursosBiblioteca[i];
      break;
    }
  }

  document.getElementById('curso-titulo').value =
    curso ? curso.titulo : '';
}

// V108: Guardar con estructura completa tipo calendario
function _guardarCurso() {

  var data = {
    id: _cursoModo === 'MODIFICAR' ? _pEvtActual.id : null,

    sigla: document.getElementById('curso-sigla').value,
    titulo: document.getElementById('curso-titulo').value,
    tipo: document.getElementById('curso-tipo').value,
    fecha: document.getElementById('curso-fecha').value,
    horaInicio: document.getElementById('curso-hora-inicio').value,
    horaFin: document.getElementById('curso-hora-fin').value,
    instructor: document.getElementById('curso-instructor').value,
    modalidad: document.getElementById('curso-modalidad').value,
    participantes: document.getElementById('curso-participantes').value,
    sala: document.getElementById('curso-sala').value,
    dia: document.getElementById('curso-dia').value,
    notas: document.getElementById('curso-notas').value
  };

  _planToast('Guardando...', 'info');

  google.script.run
    .withSuccessHandler(function(r) {
      if (r && r.success) {
        _planToast('Curso guardado', 'ok');
        _cerrarModalCurso();
        loadPlanEvents(true);
      } else {
        _planToast('Error: ' + (r ? r.error : ''), 'err');
      }
    })
    .withFailureHandler(function(e) {
      _planToast('Error: ' + e.message, 'err');
    })

    .guardarCursoPlan(data);
}

function _fmtHora(v) {
      if (!v) return '';
      var p = v.split(':');

}

// CANCELAR CURSO
function _cancelarCursoActual() {
  if (!_pEvtActual) return;

  _planPromptShow(
    'Motivo de cancelación (obligatorio):',
    function(motivo) {

      _planToast('Cancelando curso...', 'info');

      google.script.run
        .withSuccessHandler(function(r) {
          if (r && r.success) {
            _planToast('Curso cancelado', 'ok');
            closePlanEvt();
            loadPlanEvents(true);
          } else {
            _planToast('Error: ' + (r ? r.error : ''), 'err');
          }
        })
        .withFailureHandler(function(e) {
          _planToast('Error: ' + e.message, 'err');
        })
        .cancelarCursoPlan(_pEvtActual.id, motivo);
    },
    {
      title: 'Cancelar curso',
      confirmText: 'Cancelar curso'
    }
  );
}
