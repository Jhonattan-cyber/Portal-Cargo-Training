// ═══ KPI STRIP ═══
function _renderKPIPlaceholders() {
  var items = [
    { id: 'activos', icon: 'ph-users-three', label: 'Activos', color: '#3b82f6' },
    { id: 'vencidos', icon: 'ph-warning-circle', label: 'Vencidos', color: '#ef4444' },

    { id: 'vigentes', icon: 'ph-check-circle', label: 'Vigentes', color: '#22c55e' },
    { id: 'cumplimiento', icon: 'ph-chart-line-up', label: 'Cumplimiento', color: '#8b5cf6' }
  ];
  var h = '';
  for (var i = 0; i < items.length; i++) {
    var it = items[i];
    h += '<div class="ct2-kpi-card" id="ct2-kpi-' + it.id + '" style="--kpi-color:' + it.color + ';animation-delay:' + (i * 0.08) + 's;" data-testid="ct-kpi-' + it.id + '">';

function _ctUpdateKPIs(kpis) {
  if (!kpis) return;
  var map = { activos: kpis.activos, vencidos: kpis.vencidos, porVencer: kpis.porVencer, vigentes: kpis.vigentes, cumplimiento: kpis.cumplimiento + '%' };
  var keys = Object.keys(map);
  for (var i = 0; i < keys.length; i++) {
    var el = document.getElementById('ct2-kpiv-' + keys[i]);
    if (el) {
      el.textContent = _formatNum(map[keys[i]]);
      el.parentElement.classList.add('ct2-kpi-loaded');
    }
  }
  _ctKPIsLoaded = true;
}

function _formatNum(v) {
  if (typeof v === 'string') return v;
  if (v >= 1000) return (v / 1000).toFixed(1) + 'k';
  return String(v);
}

// ═══ AUTO-LOAD DASHBOARD ═══
function _ctAutoLoadDashboard() {
  google.script.run.withSuccessHandler(function(r) {
    if (r && r.success) {
      _ctUpdateKPIs(r.kpis);
      var welcomeEl = document.getElementById('ct2-welcome-text');
      if (welcomeEl && r.summary) {
        welcomeEl.innerHTML = _formatAIText(r.summary);
      } else if (welcomeEl) {
        welcomeEl.innerHTML = 'Datos cargados. Escribe tu consulta o selecciona una opcion.';
      }
    } else {
      // KPIs-only fallback
      google.script.run.withSuccessHandler(function(r2) {
        if (r2 && r2.success) _ctUpdateKPIs(r2.kpis);
      }).getControlTowerKPIs(_ctBase);
      var welcomeEl = document.getElementById('ct2-welcome-text');
      if (welcomeEl) welcomeEl.innerHTML = 'Listo. Escribe tu consulta o selecciona una opcion rapida.';
    }
  }).withFailureHandler(function(e) {
    var welcomeEl = document.getElementById('ct2-welcome-text');
    if (welcomeEl) welcomeEl.innerHTML = 'Listo. Escribe tu consulta o selecciona una opcion rapida.';
    google.script.run.withSuccessHandler(function(r2) {
      if (r2 && r2.success) _ctUpdateKPIs(r2.kpis);
    }).getControlTowerKPIs(_ctBase);
  }).getControlTowerAutoSummary(_ctBase);
}

// ═══ SEND / QUICK ═══
function ctSend() {
  var input = document.getElementById('ct2-input');
  var txt = (input.value || '').trim();
  if (!txt || _ctLoading) return;
  input.value = '';
  _ctDoSend(txt);
}

function ctQuick(key) {
  if (_ctLoading) return;
  var preguntas = {
    'inconsistencias': 'Analiza inconsistencias de datos',
    'top_vencidos': 'Top jefes con mas vencidos',
    'sin_data': 'Usuarios sin data',
    'resumen': 'Resumen general',
    'criticos': 'Criticos por base',
    'glosario': 'Glosario de siglas',
    'ia_libre': 'Analiza la situacion actual y dame recomendaciones de accion prioritarias'
  };
  _ctDoSend(preguntas[key] || key);
}

function _ctDoSend(pregunta) {
  _ctLoading = true;
  var msgs = document.getElementById('ct2-messages');
  if (!msgs) return;

  // User bubble
  var userDiv = document.createElement('div');
  userDiv.className = 'ct2-user-msg ct2-fade-in';
  userDiv.innerHTML = '<div class="ct2-user-bubble">' + escapeHtml(pregunta) + '</div>';
  msgs.appendChild(userDiv);

  // Typing indicator
  var typing = document.createElement('div');
  typing.className = 'ct2-ai-typing ct2-fade-in';
  typing.id = 'ct2-typing';
  typing.innerHTML = '<div class="ct2-ai-avatar ct2-ai-avatar-sm"><i class="ph ph-brain"></i></div>'
    + '<div class="ct2-typing-bubble"><div class="ct2-typing-inline"><span></span><span></span><span></span></div>'
    + '<span class="ct2-typing-label">Analizando...</span></div>';
  msgs.appendChild(typing);
  ctScrollDown();

  google.script.run.withSuccessHandler(function(resp) {
    _ctLoading = false;
    var t = document.getElementById('ct2-typing');
    if (t) t.remove();

    if (resp && resp.error) {
      ctAddBotMsg('<div style="color:#ef4444;"><i class="ph ph-warning" style="margin-right:4px;"></i>' + escapeHtml(resp.error) + '</div>');
    } else if (resp && resp.tipo === 'ayuda') {
      ctRenderAyuda(resp);
    } else if (resp && resp.tipo === 'ia_response') {
      ctRenderIA(resp);
    } else if (resp && resp.tipo) {
      _ctLastReport = resp;
      ctRenderReport(resp);
    } else {
      ctAddBotMsg('Respuesta inesperada.');
    }
  }).withFailureHandler(function(e) {
    _ctLoading = false;
    var t = document.getElementById('ct2-typing');
    if (t) t.remove();
    ctAddBotMsg('<div style="color:#ef4444;">Error: ' + escapeHtml(e.message) + '</div>');
  }).botAskQuestion(pregunta, _ctBase);
}

// ═══ RENDER: Structured Report ═══
function ctRenderReport(resp) {
  var h = '<div class="ct2-card ct2-fade-in">';
  h += '<div class="ct2-card-header">';
  h += '<div class="ct2-card-icon"><i class="ph ph-chart-bar"></i></div>';
  h += '<div><div class="ct2-card-title">' + escapeHtml(resp.titulo || 'Reporte') + '</div>';
  if (resp.subtitulo) h += '<div class="ct2-card-sub">' + escapeHtml(resp.subtitulo) + '</div>';
  h += '</div></div>';
  h += '<div class="ct2-card-body">';

  // KPIs
  if (resp.kpis && resp.kpis.length) {
    h += '<div class="ct2-report-kpis">';
    for (var ki = 0; ki < resp.kpis.length; ki++) {
      var kp = resp.kpis[ki];
      h += '<div class="ct2-report-kpi"><div class="ct2-report-kpi-val">' + escapeHtml(String(kp.valor)) + '</div>';
      h += '<div class="ct2-report-kpi-lbl">' + escapeHtml(kp.label) + '</div></div>';
    }
    h += '</div>';
  }

  // Table
  if (resp.tabla && resp.tabla.length) {
    h += '<div class="ct2-table-wrap"><table class="ct2-table"><thead><tr>';
    var cols = Object.keys(resp.tabla[0]);
    for (var ci = 0; ci < cols.length; ci++) h += '<th>' + escapeHtml(cols[ci]) + '</th>';
    h += '</tr></thead><tbody>';
    for (var ri = 0; ri < resp.tabla.length; ri++) {
      h += '<tr>';
      for (var ci2 = 0; ci2 < cols.length; ci2++) h += '<td>' + escapeHtml(String(resp.tabla[ri][cols[ci2]] || '')) + '</td>';
      h += '</tr>';
    }
    h += '</tbody></table></div>';
  }

  // Alertas
  if (resp.alertas && resp.alertas.length) {
    for (var ai = 0; ai < resp.alertas.length; ai++) {
      h += '<div class="ct2-alert"><i class="ph ph-warning" style="color:#f59e0b;margin-right:6px;"></i>' + escapeHtml(resp.alertas[ai]) + '</div>';
    }
  }

  h += '</div>';
  if (resp.evidencia) h += '<div class="ct2-card-footer">' + escapeHtml(resp.evidencia) + '</div>';

  // Export button
  h += '<div class="ct2-card-actions">';
  h += '<button onclick="ctExport()" class="ct2-btn-export" data-testid="ct-export"><i class="ph ph-export" style="margin-right:4px;"></i>Exportar</button>';
  h += '</div>';
  h += '</div>';

  var msgs = document.getElementById('ct2-messages');
  if (!msgs) return;
  var div = document.createElement('div');
  div.className = 'ct2-ai-msg ct2-fade-in';
  div.innerHTML = '<div class="ct2-ai-avatar ct2-ai-avatar-sm"><i class="ph ph-chart-bar"></i></div>' + h;
  msgs.appendChild(div);
  ctScrollDown();
}

// ═══ RENDER: Ayuda / Greeting ═══
function ctRenderAyuda(resp) {
  var h = '<div class="ct2-card ct2-fade-in">';
  h += '<div class="ct2-card-body">';
  if (resp.titulo) h += '<div style="font-weight:700;font-size:13px;color:#1e293b;margin-bottom:6px;">' + escapeHtml(resp.titulo) + '</div>';
  if (resp.mensaje) h += '<div style="font-size:12px;color:#64748b;margin-bottom:10px;">' + escapeHtml(resp.mensaje) + '</div>';
  if (resp.sugerencias && resp.sugerencias.length) {
    h += '<div style="display:flex;flex-wrap:wrap;gap:6px;">';
    for (var si = 0; si < resp.sugerencias.length; si++) {
      var s = resp.sugerencias[si];
      h += '<button onclick="_ctDoSend(\'' + escapeHtml(s.query).replace(/'/g, "\\'") + '\')" class="ct2-chip" style="font-size:11px;">' + escapeHtml(s.label) + '</button>';
    }
    h += '</div>';
  }
  h += '</div></div>';
  ctAddBotMsg(h);
}

// ═══ FORMAT AI TEXT ═══
function _formatAIText(txt) {
  if (!txt) return '';
  txt = txt.replace(/</g, '&lt;').replace(/>/g, '&gt;');
  txt = txt.replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>');
  txt = txt.replace(/^[\-\*]\s+(.+)/gm, '<li>$1</li>');
  txt = txt.replace(/^\d+\.\s+(.+)/gm, '<li>$1</li>');
  txt = txt.replace(/\n/g, '<br>');
  return txt;
}

// ═══ UTILITIES ═══
function _ctChangeBase(val) { _ctBase = val; _ctKPIsLoaded = false; _ctAutoLoadDashboard(); }
function ctScrollDown() { var el = document.getElementById('ct2-messages'); if (el) el.scrollTop = el.scrollHeight; }
function ctClearChat() {
  var msgs = document.getElementById('ct2-messages');
  if (msgs) msgs.innerHTML = '';
  _ctHistory = [];
  _ctLastReport = null;
  _ctAutoLoadDashboard();
}
function ctExport() {
  if (!_ctLastReport) return;
  google.script.run.withSuccessHandler(function(r) {
    if (r && r.url) window.open(r.url, '_blank');
  }).botExportToSheet(_ctLastReport);
}
function escapeHtml(str) {
  if (!str) return '';
  return String(str).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');
}
