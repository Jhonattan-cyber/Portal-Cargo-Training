/**
 * V92: Obtiene el estado del toggle de recordatorios.
 * @returns {boolean} true si la funcion de recordatorios esta habilitada
 */
function getRecordatorioEnabled() {
  try {
    var val = PropertiesService.getScriptProperties().getProperty('REMINDER_FEATURE_ENABLED');
    return val === 'true';
  } catch(e) { return false; }
}

/**
 * V92: Activa/desactiva la funcion de recordatorios (solo Admin).
 * @param {boolean} enabled - true para activar, false para desactivar
 * @returns {Object} {success, enabled}
 */
function setRecordatorioEnabled(enabled) {
  var usr = getUsuarioActual();
  if (!usr || (usr.rol || '').toUpperCase() !== 'ADMIN') {
    return { success: false, error: 'Solo ADMIN puede cambiar esta configuracion.' };
  }
  var val = enabled ? 'true' : 'false';
  PropertiesService.getScriptProperties().setProperty('REMINDER_FEATURE_ENABLED', val);
  Logger.log('[V92-TOGGLE] Recordatorio cambiado a ' + val + ' por ' + (usr.email || 'unknown'));
  return { success: true, enabled: enabled === true };
}

/**
 * Envía correos de recordatorio a colaboradores con cursos pendientes.
 * @param {Array} personas - Array de {correo, nombre, cursosPendientes: [{nombre, status, vencimiento}]}
 * @param {string} modo - 'individual' o 'grupal'
 * @param {string} nombreJefe - Nombre del jefe que envía
 * @return {Object} {success, enviados, errores}
 */
function enviarRecordatorioCursos(personas, modo, nombreJefe) {
  if (!personas || personas.length === 0) {
    return { success: false, error: 'No hay personas seleccionadas.' };
  }

  var GROUP_FROM = 'grp_entrenamientos_cgo-grh@latam.com';
  var GROUP_NAME = 'Entrenamientos CGO-GRH';
  var appUrl = '';
  try { appUrl = ScriptApp.getService().getUrl(); } catch(e) {}

  var inlineImages = {};
  try {
    var logoBlob = Utilities.newBlob(Utilities.base64Decode(LOGO_BASE64), 'image/png', 'logo.png');
    inlineImages['logo_latam'] = logoBlob;
  } catch(e) {}

  var enviados = 0;
  var errores = [];

  if (modo === 'grupal') {
    // Un solo correo con todos los destinatarios
    var emailList = personas.map(function(p) { return (p.correo || '').trim(); }).filter(function(e) { return e.length > 0; });
    if (emailList.length === 0) return { success: false, error: 'No hay correos validos.' };

    var htmlBody = _buildRecordatorioGrupal(personas, nombreJefe, appUrl);
    var asunto = 'Recordatorio de Cursos Pendientes - Entrenamientos CGO-GRH';

    var optsBase = { htmlBody: htmlBody, inlineImages: inlineImages, name: GROUP_NAME, replyTo: GROUP_FROM };
    try {
      GmailApp.sendEmail(emailList.join(','), asunto, ' ', Object.assign({}, optsBase, { from: GROUP_FROM }));
      enviados = emailList.length;
    } catch(e1) {
      try {
        GmailApp.sendEmail(emailList.join(','), asunto, ' ', optsBase);
        enviados = emailList.length;
      } catch(e2) { errores.push('Error enviando grupal: ' + e2.message); }
    }
    registrarEnvioCorreo(asunto, emailList.join(', '), enviados > 0 ? 'OK (' + enviados + ')' : 'ERROR');

  } else {
    // Correos individuales
    for (var i = 0; i < personas.length; i++) {
      var p = personas[i];
      var email = (p.correo || '').trim();
      if (!email) { errores.push('Sin correo: ' + (p.nombre || '?')); continue; }

      var htmlBody = _buildRecordatorioIndividual(p, nombreJefe, appUrl);
      var asunto = 'Recordatorio: Cursos Pendientes - ' + (p.nombre || '');

      var optsBase = { htmlBody: htmlBody, inlineImages: inlineImages, name: GROUP_NAME, replyTo: GROUP_FROM };
      try {
        GmailApp.sendEmail(email, asunto, ' ', Object.assign({}, optsBase, { from: GROUP_FROM }));
        enviados++;
        registrarEnvioCorreo(asunto, email, 'OK');
      } catch(e1) {
        try {
          GmailApp.sendEmail(email, asunto, ' ', optsBase);
          enviados++;
          registrarEnvioCorreo(asunto, email, 'OK (fallback)');
        } catch(e2) {
          errores.push(p.nombre + ': ' + e2.message);
          registrarEnvioCorreo(asunto, email, 'ERROR: ' + e2.message);
        }
      }
    }
  }

  return {
    success: enviados > 0,
    enviados: enviados,
    total: personas.length,
    errores: errores,
    message: 'Se enviaron ' + enviados + ' de ' + personas.length + ' recordatorios.'
  };
}

function _buildRecordatorioGrupal(personas, nombreJefe, appUrl) {

    var cursos = p.cursosPendientes || [];
    var cursosTexto = cursos.map(function(c) {
      var label = c.status === 'VENCIDO' ? 'Vencido' : 'Prox. a vencer';
      var dot = c.status === 'VENCIDO' ? '&#x25CF;' : '&#x25CF;';
      var color = c.status === 'VENCIDO' ? '#dc2626' : '#d97706';
      return '<span style="color:' + color + ';">' + dot + '</span> ' + (c.nombre || c.sigla || '-') + ' <span style="color:#94a3b8;">(' + label + ' - ' + (c.vencimiento || '-') + ')</span>';
    }).join('<br>');
    var peorStatus = cursos.some(function(c) { return c.status === 'VENCIDO'; }) ? 'VENCIDO' : 'PROGRAMAR';
    var bgColor = peorStatus === 'VENCIDO' ? '#fef2f2' : '#fffbeb';

    filasHTML += '<tr style="border-bottom:1px solid #e2e8f0;">' +
      '<td style="padding:12px 16px;font-size:14px;color:#1e293b;font-weight:600;background:' + bgColor + ';vertical-align:top;">' + (p.nombre || '-') + '</td>' +
      '<td style="padding:12px 16px;font-size:12px;color:#475569;line-height:1.6;background:' + bgColor + ';">' + cursosTexto + '</td>' +
