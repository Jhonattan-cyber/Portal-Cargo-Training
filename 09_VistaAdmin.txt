/**
 * Obtiene estudiantes de una base específica para vista Admin
 * @param {string} tipoBase - 'LATAM_SUR', 'CL', 'RAMPA', 'NAM_EU'
 */
function getStudentsFromBase(tipoBase) {
  try {
    var baseKey = tipoBase;
    // Mapear nombres de vista a keys internos
    if (tipoBase === 'LATAM_SUR') baseKey = 'LATAM';
    else if (tipoBase === 'CL') baseKey = 'CHILE';
    else if (tipoBase === 'NAM_EU') baseKey = 'NAM';
    
    var empleados = leerHoja(baseKey, 'base');
    
    // Transformar a formato Admin
    var students = empleados.map(function(e) {
      return {
        bp: e.bp, nombre: e.nombre, correo: e.correo, pais: e.pais,
        area: e.area, jefeDirecto: e.jefeDirecto, jefeArea: e.jefeArea,
        cursosActivos: e.cursosAplicables || e.totalCursos,
        cursosHabilitados: e.habilitados, cursosPorVencer: e.porVencer,
        cursosVencidos: e.vencidosAplicables || e.vencidos,
        cumplimiento: e.cumplimiento, semaforo: e.semaforo,
        habilitacionGeneral: e.habilitacionGeneral, tieneAlerta: e.tieneAlerta,
        esNuevoIngreso: e.esNuevoIngreso || false
      };
    });
    
    return { students: students, total: students.length };
  } catch(e) {
    return { error: e.message, students: [] };
  }
}

/**
 * Obtiene estudiantes HCC de todas las bases
 * HCC = Area contiene "HCC"
 */
function getStudentsHCCFromAllBases() {
  try {
    var allStudents = [];
    
    ['CHILE', 'RAMPA', 'LATAM', 'NAM'].forEach(function(bk) {
      try {
        var empleados = leerHoja(bk, 'base');
        empleados.forEach(function(e) {
          // Filtrar solo HCC (detección multi-columna)
          if (e.esHCC) {
            allStudents.push({
              bp: e.bp, nombre: e.nombre, correo: e.correo, pais: e.pais,
              area: e.area, jefeDirecto: e.jefeDirecto, jefeArea: e.jefeArea,
              cursosActivos: e.cursosAplicables || e.totalCursos,
              cursosHabilitados: e.habilitados, cursosPorVencer: e.porVencer,
              cursosVencidos: e.vencidosAplicables || e.vencidos,
              cumplimiento: e.cumplimiento, semaforo: e.semaforo,
              habilitacionGeneral: e.habilitacionGeneral, tieneAlerta: e.tieneAlerta,
              baseOriginal: bk,
              esNuevoIngreso: e.esNuevoIngreso || false
            });
          }
        });
      } catch(e) { Logger.log('Error HCC ' + bk + ': ' + e.message); }
    });
    
    return { students: allStudents, total: allStudents.length };
  } catch(e) {
    return { error: e.message, students: [] };
  }
}

/**

  _log(2, '[ARG-AUDIT] argsRaw: tipoBase=' + rawTipoBase + ' dias=' + rawDias + ' fuente=' + fuente);

  var t0ByBase = new Date().getTime();
  try {
    // ═══ FIX B: Si usuarioInfo está vacío, recuperar de sesión ═══
    if (!usuarioInfo || !usuarioInfo.nombre || !usuarioInfo.email) {
      var _usr = getUsuarioActual();
      if (_usr) {
        usuarioInfo = {
          nombre: _usr.nombre || '',
          email: _usr.email || '',

    if (usuarioBase === 'TRAINING CL' || usuarioBase === 'CL') usuarioBase = 'CHILE';
    if (usuarioBase === 'PHH RED' || usuarioBase === 'PHH') usuarioBase = 'LATAM';
    if (usuarioBase === 'CTRL NORTE' || usuarioBase === 'CONTROL NORTE') usuarioBase = 'NAM';

      'HCC': 'HCC',
      'LATAM_SUR': 'LATAM',
      'CL': 'CHILE',
      'CHILE': 'CHILE',
      'RAMPA': 'RAMPA',
      'NAM_EU': 'NAM',
      'NAM': 'NAM',
      'LATAM': 'LATAM',
      'TODO': 'TODO'
    };
    var filtro = filtroMap[tipoBase.toUpperCase()] || tipoBase.toUpperCase();
    
    Logger.log('[ARG-AUDIT] resolved: tipoBase="' + tipoBase + '" → filtro="' + filtro + '" | usuarioBase="' + usuarioBase + '"');

    // ═══ V50-FIX: Declarar rolUsuario ANTES de su primer uso (antes era hoisting bug) ═══
    var rolUsuario = (usuarioInfo && usuarioInfo.role || '').toUpperCase();
    var nombreUsuarioInfo = (usuarioInfo && usuarioInfo.nombre || '').trim();

    // V48: JEFES — leer TODAS las bases para que BPs multi-base tengan registros de cursos
    // de ambas fuentes. El filtro por equipoBPs (línea 5150) se encarga de limitar el scope.
    // V49: ADMIN/CAPACITY también leen TODAS las bases cuando fuente=CONSOLIDADO
    var filtroParaData = filtro;
    if (filtro !== 'TODO' && filtro !== 'HCC') {
      if (rolUsuario !== 'ADMIN' && rolUsuario !== 'CAPACITY') {
        filtroParaData = 'TODO';

      }
      // V96: PLAN_SINGLE = solo leer la base especifica (para busqueda en Planificacion)
    }
    _log('  Llamando getForecastData("' + filtroParaData + '", usuarioBase="' + usuarioBase + '")...');
    var data = getForecastData(filtroParaData, usuarioBase);
    // V100-CLEAN: Log VISIBLE de getForecastData resultado
    _auditLog('[V100-FC-DATA] filtro=' + filtro + ' filtroParaData=' + filtroParaData + ' | registrosRaw=' + (data.registros ? data.registros.length : 0) + ' success=' + data.success);
    _log('  getForecastData retornó: success=' + data.success + ', registros=' + (data.registros ? data.registros.length : 0) + ' (filtroParaData=' + filtroParaData + ') | ' + (new Date().getTime() - t0ByBase) + 'ms');
    if (!data.success) {
      Logger.log('◀◀◀ END getForecastDataByBase("' + tipoBase + '") [ERROR en getForecastData] ' + (data.error || ''));
      return { error: data.error || 'getForecastData falló', forecast: [] };
    }
    
    // EXPAND: Minified keys -> full names for frontend compatibility
    // V50: Normalizar BP (trim) para evitar duplicados por whitespace
    var forecast = data.registros.map(function(r) {
      return {
        bp: String(r.b || '').trim(),
        nombre: r.n,
        pais: r.p,
        correo: r.co,
        jefeDirecto: r.jd   || '',      // SUPERVISOR   (nivel 7)
        jefeArea:    r.ja   || '',      // JEFE AREA    (nivel 6)
        jefeDepartamento: r.jdpto || '', // JEFE DEPTO  (nivel 5)
        subgerente:  r.sg   || '',      // SUBGERENTE   (nivel 4)
        gerente:     r.g    || '',      // GERENTE      (nivel 3)
        gerenteSr:   r.gsr  || '',      // GERENTE SR   (nivel 2)
        director:    r.dir  || '',      // DIRECTOR     (nivel 1)

        negocio: r.neg || '',
        zona: r.z || '',
        areaCanon: r.ac || '',
        posicion: r.pos || '',

        curso: r.cu,

        fechaVencimiento: r.v,
        habilitacion: r.h,
        diasRestantes: r.d,
        status: r.s,
        statusReal: r.sr || r.s,

        esHCC: r.hcc || false,

    });

    // ═══════════════════════════════════════════════════════════════════════
    // V50: DEDUP POR BP — Registros de cursos + Personas únicas
    // Cuando un BP aparece en múltiples bases (PHH + RAMPA), hay registros
    // duplicados de cursos. La regla de merge:
    //   - RAMPA TIENE PRIORIDAD sobre PHH para metadata (pais, area, jerarquía)
    //   - Se conservan TODOS los registros de cursos (sin dedup de cursos)
    //   - La dedup es solo de PERSONAS para conteos correctos
    // ═══════════════════════════════════════════════════════════════════════
    var _beforeMerge = forecast.length;
    var _bpSeenBases = {}; // { bp: { bases: ['LATAM','RAMPA'], prioBase: 'RAMPA' } }
    forecast.forEach(function(r) {
      if (!r.bp) return;
      if (!_bpSeenBases[r.bp]) {
        _bpSeenBases[r.bp] = { bases: [r.base], prioBase: r.base };
      } else {
        if (_bpSeenBases[r.bp].bases.indexOf(r.base) < 0) {
          _bpSeenBases[r.bp].bases.push(r.base);
        }
        // RAMPA tiene prioridad sobre otros para metadata
        if (r.base === 'RAMPA') _bpSeenBases[r.bp].prioBase = 'RAMPA';
      }
    });
    // Contar personas únicas ANTES de jerarquía
    var _uniqueBPsBeforeJerarquia = Object.keys(_bpSeenBases).length;
    // Identificar BPs duplicados (multi-base)
    var _dupBPs = [];
    for (var _dbp in _bpSeenBases) {
      if (_bpSeenBases[_dbp].bases.length > 1) _dupBPs.push(_dbp);
    }
    // Para BPs duplicados: normalizar metadata usando la base prioritaria (RAMPA > otros)
    if (_dupBPs.length > 0) {
      // Construir mapa de metadata prioritaria por BP
      var _prioMeta = {};
      forecast.forEach(function(r) {
        if (!r.bp || _dupBPs.indexOf(r.bp) < 0) return;
        var prio = _bpSeenBases[r.bp].prioBase;
        if (r.base === prio && !_prioMeta[r.bp]) {
          _prioMeta[r.bp] = {
            pais: r.pais, area: r.area, negocio: r.negocio, areaCanon: r.areaCanon, zona: r.zona,
            jefeDirecto: r.jefeDirecto, jefeArea: r.jefeArea,
            jefeDepartamento: r.jefeDepartamento, subgerente: r.subgerente,
            gerente: r.gerente, gerenteSr: r.gerenteSr, director: r.director
          };
        }
      });
      // Aplicar metadata prioritaria a TODOS los registros de BPs duplicados
      forecast.forEach(function(r) {
        if (_prioMeta[r.bp]) {
          var m = _prioMeta[r.bp];
          r.pais = m.pais; r.area = m.area; r.negocio = m.negocio; r.areaCanon = m.areaCanon;
          r.jefeDirecto = m.jefeDirecto; r.jefeArea = m.jefeArea;
          r.jefeDepartamento = m.jefeDepartamento; r.subgerente = m.subgerente;

        }
      });
      _log('[V0-DEDUP] BPs duplicados (multi-base): ' + _dupBPs.length + ' → metadata normalizada con prioridad RAMPA');

    }
    Logger.log('[FORECAST] beforeMerge=' + _beforeMerge + ' afterMergeUniqueBP=' + _uniqueBPsBeforeJerarquia);
    
    // ═══════════════════════════════════════════════════════════════════════
    // V51: Construir mapa nombre→BP para resolver jefeDirectoBP
    // Cada empleado tiene nombre y BP. Buscamos el BP del jefe por su nombre.
    // ═══════════════════════════════════════════════════════════════════════
    var _nombreBPMap = {};  // { normNombre: bp }
    forecast.forEach(function(r) {
      if (r.bp && r.nombre) {
        var nn = normNombre(r.nombre);
        if (nn && !_nombreBPMap[nn]) _nombreBPMap[nn] = r.bp;
      }
    });
    // Asignar jefeDirectoBP a cada registro — resolución exhaustiva
    var _jdResolved = 0, _jdTotal = 0, _jdAmbiguous = 0;
    forecast.forEach(function(r) {
      if (r.jefeDirecto) {
        _jdTotal++;
        var jdNorm = normNombre(r.jefeDirecto);
        var bp = _nombreBPMap[jdNorm];
        
        // PASO 1: Match exacto — si existe, usar directamente (sin fuzzy)
        if (bp) {
          r.jefeDirectoBP = bp;
          _jdResolved++;
          return; // DONE — no usar fuzzy
        }
        
        // PASO 2: Fuzzy fallback SOLO si no hay match exacto
        var candidates = [];
        var tokens = jdNorm.split(/\s+/).filter(function(p) { return p.length > 1; });
        if (tokens.length >= 2) {
          for (var nk2 in _nombreBPMap) {
            var allFound = true;
            for (var tk = 0; tk < tokens.length; tk++) {
              if (nk2.indexOf(tokens[tk]) < 0) { allFound = false; break; }
            }
            if (allFound) candidates.push({ name: nk2, bp: _nombreBPMap[nk2] });
          }
        }
        
        if (candidates.length === 1) {
          // Fuzzy unambiguo — usar
          r.jefeDirectoBP = candidates[0].bp;
          _jdResolved++;
        } else if (candidates.length > 1) {
          // Ambiguo — loguear y NO asignar
          _jdAmbiguous++;
          _log('[V1-JEFE-BP] ambiguous name="' + r.jefeDirecto + '" candidates=[' + candidates.map(function(c) { return c.bp; }).join(',') + ']');
          r.jefeDirectoBP = '';
        } else {
          r.jefeDirectoBP = '';
        }
      } else {
        r.jefeDirectoBP = '';
      }
    });
    var _jdPct = _jdTotal > 0 ? Math.round((_jdResolved / _jdTotal) * 100) : 100;
    _log('[V1-JEFE-BP] nombreBPMap=' + Object.keys(_nombreBPMap).length + ' | resuelto: ' + _jdResolved + '/' + _jdTotal + ' (' + _jdPct + '%) | ambiguos: ' + _jdAmbiguous);
    
    // ═══════════════════════════════════════════════════════════════════════
    // V22c: FILTRO JERÁRQUICO UNIFICADO usando buildArbolJerarquico
    // ═══════════════════════════════════════════════════════════════════════
    var forecastAntes = forecast.length;
    // V50: rolUsuario y nombreUsuarioInfo ya declarados arriba (fix hoisting)
    var areaUsuarioInfo = (usuarioInfo && usuarioInfo.area || '').trim();
    var emailUsuarioInfo = (usuarioInfo && usuarioInfo.email || 'N/A');
    var nk = normNombre(nombreUsuarioInfo);
    var nf = normalizarNombreJefe(nombreUsuarioInfo);
    
    Logger.log('═══════ FILTRO JERARQUÍA V50 ═══════');
    Logger.log('DBG filtroJerarquia: email=' + emailUsuarioInfo + ' nombre="' + nombreUsuarioInfo + '" rol=' + rolUsuario);
    Logger.log('DBG areaUsuario=' + areaUsuarioInfo + ' (NO se usa para limitar arbol, solo metadata)');
    _log('  email: ' + emailUsuarioInfo + ' | nombre: "' + nombreUsuarioInfo + '" | rol: ' + rolUsuario);
    _log('  registros TOTAL dataset: ' + forecastAntes + ' | personas únicas (pre-jerarquía): ' + _uniqueBPsBeforeJerarquia);
    
    // V44: Initialize shared variables used after ALL conditional branches
    var personasSet = {};
    var jefesSet = {};
    var jefesAreaSet = {};
    var equipoTotalActivos = 0;
    var totalNuevosIngresosArbol = 0;
    var niBPsList = [];
    var arbol = null;
    var equipoBPs = {};
    var miCadena = { director:'', gerenteSr:'', gerente:'', subgerente:'', jefeDepartamento:'', jefeArea:'', supervisor:'' };
    var nivelAgrupacion = '';
    var campoAgrupacion = '';
    var filaUsuario = null;
    
    // V52: CAPACITY users — vista consolidada como ADMIN, con filtro geográfico opcional
    if (rolUsuario === 'CAPACITY') {
      var _capT0 = new Date().getTime();
      Logger.log('[V76-CAPACITY] >>> START procesando CAPACITY: email=' + emailUsuarioInfo + ' filtro=' + filtro + ' fuente=' + fuente + ' filtroParaData=' + filtroParaData + ' forecast.length=' + forecast.length);
      var capFilter = (usuarioInfo && usuarioInfo.capacityFilter) || null;
      // V52: Log obligatorio [CAPACITY-DBG]
      var _capBasesEnDatos = {};
      forecast.forEach(function(r) { _capBasesEnDatos[r.base || 'SIN_BASE'] = (_capBasesEnDatos[r.base || 'SIN_BASE'] || 0) + 1; });
      Logger.log('[CAPACITY-DBG] rol=CAPACITY basesLeidas=' + JSON.stringify(Object.keys(_capBasesEnDatos)) + ' registrosPorBase=' + JSON.stringify(_capBasesEnDatos) + ' totalRegistros=' + forecast.length + ' capFilter=' + JSON.stringify(capFilter));

      if (capFilter && ((capFilter.allowedCountries || []).length > 0 || (capFilter.excludedAreas || []).length > 0)) {
        // Mapa amplio: normKey() quita acentos (PERÚ→PERU, PANAMÁ→PANAMA)
        var PAIS_NORMALIZE = {
          'COLOMBIA': 'CO', 'PERU': 'PE', 'ECUADOR': 'EC', 'CHILE': 'CL',
          'UNITED STATES': 'US', 'US': 'US', 'USA': 'US', 'ESTADOS UNIDOS': 'US',
          'ARGENTINA': 'AR', 'URUGUAY': 'UY', 'PARAGUAY': 'PY',
          'BRASIL': 'BR', 'BRAZIL': 'BR', 'MEXICO': 'MX', 'PANAMA': 'PA',
          'COSTA RICA': 'CR', 'REPUBLICA DOMINICANA': 'DO', 'GUATEMALA': 'GT',
          'HONDURAS': 'HN', 'EL SALVADOR': 'SV', 'NICARAGUA': 'NI',
          'BOLIVIA': 'BO', 'VENEZUELA': 'VE', 'PUERTO RICO': 'PR'
        };

        (capFilter.allowedCountries || []).forEach(function(c) { countriesAllowed[c.toUpperCase()] = true; });
        var excAreasNorm = (capFilter.excludedAreas || []).map(function(a) { return normKey(a); });

        var antesCapFilter = forecast.length;
        if (forecast.length > 0) {
          var muestra = forecast.slice(0, 5).map(function(r) { return '{pais:"' + (r.pais||'') + '", area:"' + (r.area||'') + '", base:"' + (r.base||'') + '"}'; });
          Logger.log('[CAPACITY-DBG] Muestra pre-filtro: ' + muestra.join(', '));
        }

        forecast = forecast.filter(function(r) {
          // Filtro por país (normKey quita acentos)
          var paisNorm = normKey(r.pais || '');
          var paisCode = PAIS_NORMALIZE[paisNorm] || paisNorm;
          if ((capFilter.allowedCountries || []).length > 0 && !countriesAllowed[paisCode] && !countriesAllowed[paisNorm]) return false;
          // Filtro por área excluida (indexOf: HCC matchea "HCC", "HCC BOGOTA", etc.)
          if (excAreasNorm.length > 0) {
            var areaNorm = normKey(r.area || '');
            for (var eai = 0; eai < excAreasNorm.length; eai++) {
              if (areaNorm.indexOf(excAreasNorm[eai]) >= 0) return false;
            }
          }
          return true;
        });
        // V52: Log post-filtro con detalle de bases restantes
        var _capBasesPost = {};
        forecast.forEach(function(r) { _capBasesPost[r.base || 'SIN_BASE'] = (_capBasesPost[r.base || 'SIN_BASE'] || 0) + 1; });
        Logger.log('[CAPACITY-DBG] Filtro geográfico aplicado: ' + antesCapFilter + ' → ' + forecast.length + ' registros | basesPost=' + JSON.stringify(_capBasesPost) + ' | countries=' + JSON.stringify(capFilter.allowedCountries) + ' exclAreas=' + JSON.stringify(capFilter.excludedAreas));
        if (forecast.length === 0 && antesCapFilter > 0) {
          var paisesEnDatos = {};
          data.registros.forEach(function(r) {

          });

        }

      equipoTotalActivos = 0;
      var _capT1 = new Date().getTime();
      Logger.log('[V76-CAPACITY] <<< END procesando CAPACITY | ' + (_capT1 - _capT0) + 'ms | forecast.length=' + forecast.length);
    } else if (rolUsuario === 'ADMIN') {
      // ═══════════════════════════════════════════════════════════════════════
      // V51e: ADMIN — ve todo el dataset (sin filtro jerárquico)
      // El conteo se calcula abajo desde personasSet (dedup por BP, post base-filter)
      // NO usa metadata.totalXXX (que inflaba el conteo con BPs duplicados cross-base)
      // ═══════════════════════════════════════════════════════════════════════

/**
 * Retorna la biblioteca de cursos organizada por base (lee de BIBLIOTECA FINAL)
 * Para el frontend: tabla filtrable por base (javascript.html) y mapa por base (forecast.html)
 */
function getBibliotecaFinal() {
  try {
    // Validar permisos: solo ADMIN puede acceder a Biblioteca
    var usr = getUsuarioActual();
    if (usr) {
      var rolUpper = (usr.rol || '').toUpperCase();
      if (rolUpper !== 'ADMIN') {
        Logger.log('getBibliotecaFinal: acceso denegado para rol ' + rolUpper);
        return { success: false, error: 'Acceso denegado. Solo ADMIN puede acceder a Biblioteca.', cursos: [], cursosMap: {} };
      }
    }
    
    // V21d: Construir biblioteca desde CURSOS_POR_BASE (fuente canónica), NO desde hoja BIBLIOTECA FINAL
    // Esto evita cursos fantasma, duplicados y typos
    var cursos = [];
    var totalPorBase = {};
    var cursosMap = {};
    var baseMap = { 'LATAM': 'LATAM_SUR', 'CHILE': 'CL', 'RAMPA': 'RAMPA', 'NAM': 'NAM_EU' };
    var baseKeys = ['LATAM', 'CHILE', 'RAMPA', 'NAM'];
    
    for (var bi = 0; bi < baseKeys.length; bi++) {
      var base = baseKeys[bi];
      var cursosBase = CURSOS_POR_BASE[base] || [];
      var whitelist = COURSES_BY_BASE[base] || [];
      var frontendBase = baseMap[base] || base;
      cursosMap[frontendBase] = {};
      var count = 0;
      var dedupSet = {};
      
      for (var ci = 0; ci < cursosBase.length; ci++) {
        var curso = cursosBase[ci];
        var idCanon = curso.id;
        
        // Normalizar ID por si hay alias
        var idNorm = normalizarSigla(idCanon) || idCanon;
        
        // Solo incluir si está en whitelist
        if (whitelist.indexOf(idCanon) < 0 && whitelist.indexOf(idNorm) < 0) continue;
        
        // Deduplicar por ID normalizado
        if (dedupSet[idNorm]) continue;
        dedupSet[idNorm] = true;
        
        cursos.push({ base: base, sigla: idNorm, nombre: curso.nombre || idNorm });
        cursosMap[frontendBase][idNorm] = curso.nombre || idNorm;
        count++;
      }
      totalPorBase[base] = count;
    }

    Logger.log('getBibliotecaFinal (canónica): ' + cursos.length + ' cursos, bases: ' + JSON.stringify(totalPorBase));
    return {
      success: true,
      cursos: cursos,
      cursosMap: cursosMap,
      totalPorBase: totalPorBase
    };
  } catch(e) {
    Logger.log('Error getBibliotecaFinal: ' + e.message);
    return { success: false, error: e.message, cursos: [], cursosMap: {} };
  }
}

/**
 * Retorna configuración de la app (URL base para correos)
 */
function getConfiguracion() {
  try {
    var url = ScriptApp.getService().getUrl();
    return { appUrl: url || '' };
  } catch(e) {
    return { appUrl: '', error: e.message };
  }
}
