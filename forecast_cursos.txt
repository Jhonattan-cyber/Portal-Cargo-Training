var _fccData = null;

function initForecastCursos() {
  // V2-FIX: Detección de bases robusta usando APP.usuario directamente
  var bases = _fccGetBases();
  _fccTipoBase = (bases.length > 0) ? bases[0] : 'HCC';
  _fccFuente = _fccGetFuente(_fccTipoBase);
  cargarFCCursos(_fccTipoBase);
}

function _fccGetBases() {
  // Intenta usar getBasesPermitidas si está disponible y currentUser está inicializado
  if (typeof getBasesPermitidas === 'function' && typeof currentUser !== 'undefined' && currentUser) {
    return getBasesPermitidas();
  }
  // Fallback: usar APP.usuario directamente
  var u = (typeof APP !== 'undefined' && APP.usuario) ? APP.usuario : null;
  if (!u) return ['HCC'];
  var rol = (u.rol || '').toUpperCase().replace(/\s+/g, '_');
  if (rol === 'ADMIN') return ['HCC', 'LATAM_SUR', 'CL', 'RAMPA', 'NAM_EU'];
  if (rol === 'CAPACITY') {
    var capFilter = u.capacityFilter || {};
    var bToF = { 'LATAM': 'LATAM_SUR', 'RAMPA': 'RAMPA', 'NAM': 'NAM_EU', 'CHILE': 'CL' };
    var result = [];
    (capFilter.allowedBases || []).forEach(function(b) {
      var fk = bToF[b] || b;
      if (result.indexOf(fk) < 0) result.push(fk);
    });
    return result.length > 0 ? result : ['LATAM_SUR'];
  }
  // Otros roles: mapear su base
  var base = (u.base || u.baseOrigen || '').toUpperCase();
  var bMap = { 'CHILE': 'CL', 'LATAM': 'LATAM_SUR', 'NAM': 'NAM_EU', 'RAMPA': 'RAMPA' };
  return [bMap[base] || base || 'HCC'];
}

function _fccGetFuente(tipo) {
  if (tipo === 'RAMPA') return 'RAMPA';
  if (tipo === 'LATAM_SUR') return 'PHH';
  return 'CONSOLIDADO';
}

function _fccNombreBase(tipo) {
  var m = { 'HCC': 'HCC', 'LATAM_SUR': 'LATAM PHH', 'CL': 'Chile', 'RAMPA': 'Rampa', 'NAM_EU': 'NAM & EU', 'TODO': 'Todos' };
  return m[tipo] || tipo;
}

function _fccSubBase(tipo) {
  var m = { 'HCC': 'Priorizados', 'LATAM_SUR': 'Regional', 'CL': 'Solo CL', 'RAMPA': 'RED', 'NAM_EU': 'USA, Europa', 'TODO': 'Global' };
  return m[tipo] || '';
}

function cargarFCCursos(tipoBase) {
  _fccTipoBase = tipoBase;
  _fccFuente = _fccGetFuente(tipoBase);
  _fccData = null;
  _fccFilterCurso = '';
  _fccFilterArea = '';
  _fccFilterStatus = '';

  var container = document.getElementById('fcc-content');
  if (!container) return;

  // Render skeleton with base tabs immediately
  var bases = _fccGetBases();
  var html = _fccRenderHeader(bases);
  html += '<div style="text-align:center;padding:60px 20px;"><div class="fcc-spinner"></div><p style="margin-top:16px;color:#64748b;font-size:14px;">Cargando ' + _fccNombreBase(tipoBase) + '...</p></div>';
  container.innerHTML = html;

  google.script.run
    .withSuccessHandler(function(result) {
      if (!result || !result.success) {
        var errHtml = _fccRenderHeader(bases);
        errHtml += '<div class="fcc-empty"><i class="ph ph-warning-circle"></i><p>' + (result && result.error || 'Error al cargar datos') + '</p>';
        errHtml += '<button onclick="cargarFCCursos(\'' + tipoBase + '\')" style="margin-top:12px;padding:8px 16px;background:#4f46e5;color:#fff;border:none;border-radius:8px;font-weight:700;cursor:pointer;">Reintentar</button></div>';
        container.innerHTML = errHtml;
        return;
      }
      _fccData = result;
      renderFCCursos();
    })
    .withFailureHandler(function(err) {
      var errHtml = _fccRenderHeader(bases);
      errHtml += '<div class="fcc-empty"><i class="ph ph-warning-circle"></i><p>Error: ' + (err.message || err) + '</p></div>';
      container.innerHTML = errHtml;
    })
    .getForecastDataByCourse(tipoBase, _fccFuente);
}

    var el = document.getElementById('fcc-total-colab');
    if (el) el.textContent = kpis.totalCursos;

function _fccKpi(label, value, color, icon) {
  return '<div class="fcc-kpi-card"><div class="fcc-kpi-label"><i class="ph ' + icon + '" style="color:' + color + ';margin-right:3px;"></i>' + label + '</div><div class="fcc-kpi-value" style="color:' + color + ';">' + value + '</div></div>';
}

function _fccMatrixHead(monthKeys) {
  var cols = [
    { key: 'sigla', label: 'CURSO' },
    { key: 'total', label: 'TOTAL' },
    { key: 'vencidos', label: 'VENC.' },
    { key: 'porVencer', label: 'P.VENC' }
  ];
  var html = '<thead><tr>';
  for (var i = 0; i < cols.length; i++) {
    var c = cols[i];
    var sorted = (_fccSortCol === c.key);
    var arrow = sorted ? (_fccSortAsc ? ' &#9650;' : ' &#9660;') : '';
    var style = (i === 0) ? '' : 'width:50px;';
    html += '<th style="cursor:pointer;' + style + '" onclick="_fccSort(\'' + c.key + '\')">' + c.label + (sorted ? '<span style="font-size:9px;margin-left:2px;">' + arrow + '</span>' : '') + '</th>';
  }
  // Month columns (first is _ANT, rest are rolling months)
  var anioActual = (_fccData && _fccData.anioActual) ? _fccData.anioActual : new Date().getFullYear();
  for (var mi = 0; mi < monthKeys.length; mi++) {
    var mk = monthKeys[mi];
    if (mk === '_ANT') {
      html += '<th style="width:52px;line-height:1.2;background:#991b1b;color:#fff;font-size:10px;" title="Vencidos de a&ntilde;os/meses anteriores">&lt;' + anioActual + '</th>';
    } else {
      var parts = mk.split('-');
      var mesNum = parseInt(parts[1], 10);
      var anioNum = parseInt(parts[0], 10);
      var esActual = (mesNum === (new Date().getMonth() + 1) && anioNum === new Date().getFullYear());
      var thStyle = esActual ? 'width:48px;line-height:1.2;background:#312e81;' : 'width:48px;line-height:1.2;';
      html += '<th style="' + thStyle + '">' + MESES_C[mesNum] + '<br><span style="font-size:9px;opacity:0.7;">' + parts[0] + '</span></th>';
    }
  }
  html += '</tr></thead>';
  return html;
}

function _fccMatrixRows(cursos, monthKeys) {
  if (!cursos || !cursos.length) {
    var colspan = 8 + monthKeys.length;
    return '<tr><td colspan="' + colspan + '" style="text-align:center;padding:30px;color:#94a3b8;">Sin cursos para mostrar</td></tr>';
  }
  var html = '';
  for (var i = 0; i < cursos.length; i++) {
    var c = cursos[i];
    var rowBg = c.vencidos > 0 ? 'background:#fef2f2;' : c.porVencer > 0 ? 'background:#fffbeb;' : 'background:#f0fdf4;';
    html += '<tr style="' + rowBg + '" onclick="_fccDrilldown(\'' + _fccEsc(c.sigla) + '\')" data-testid="fcc-row-' + i + '">';
    // Curso name — Nombre de BIBLIOTECA prominente, sigla como etiqueta
    var displayName = (c.nombre && c.nombre !== c.sigla) ? c.nombre : c.sigla;
    html += '<td style="background:inherit;"><div style="font-weight:700;color:#1e293b;font-size:12px;max-width:220px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;" title="' + _fccEsc(displayName) + '">' + _fccEsc(displayName) + '</div>';
    if (c.nombre && c.nombre !== c.sigla) html += '<div style="font-size:9px;color:#94a3b8;font-weight:600;">' + _fccEsc(c.sigla) + '</div>';
    html += '</td>';
    // Total
    html += '<td style="font-weight:800;color:#1e293b;">' + c.total + '</td>';
    // Vencidos
    html += '<td>' + (c.vencidos > 0 ? '<span class="fcc-badge fcc-badge-red">' + c.vencidos + '</span>' : '<span style="color:#cbd5e1;">0</span>') + '</td>';
    // Por vencer
    html += '<td>' + (c.porVencer > 0 ? '<span class="fcc-badge fcc-badge-yellow">' + c.porVencer + '</span>' : '<span style="color:#cbd5e1;">0</span>') + '</td>';
    // Month cells
    for (var mi = 0; mi < monthKeys.length; mi++) {
      var mk = monthKeys[mi];
      var md = c.meses ? c.meses[mk] : null;
      if (md && md.t > 0) {
        var cellClass = 'fcc-cell-green';
        if (md.v > 0) cellClass = 'fcc-cell-red';
        else if (md.pv > 0) cellClass = 'fcc-cell-yellow';
        var tooltip = '';
        if (mk === '_ANT') {
          var antAnio = (_fccData && _fccData.anioActual) ? _fccData.anioActual : new Date().getFullYear();
          tooltip = 'Anteriores (<' + antAnio + '): ' + md.t + ' pers.';
        } else {
          tooltip = MESES_C[parseInt(mk.split('-')[1], 10)] + ' ' + mk.split('-')[0] + ': ' + md.t + ' pers.';
        }
        if (md.v > 0) tooltip += ' (' + md.v + ' venc.)';
        if (md.pv > 0) tooltip += ' (' + md.pv + ' p.venc.)';
        if (md.h > 0) tooltip += ' (' + md.h + ' hab.)';
        html += '<td class="' + cellClass + '" title="' + tooltip + '">' + md.t + '</td>';
      } else {
        html += '<td class="fcc-cell-empty">-</td>';
      }
    }
    html += '</tr>';
  }
  return html;
}

function _fccMatrixFoot(cursos, monthKeys) {
  var totalPersonas = 0, totalVenc = 0, totalPV = 0;
  var footMeses = {};
  for (var i = 0; i < cursos.length; i++) {
    var c = cursos[i];
    totalPersonas += c.total;
    totalVenc += c.vencidos;
    totalPV += c.porVencer;
    for (var mi = 0; mi < monthKeys.length; mi++) {
      var mk = monthKeys[mi];
      var md = c.meses ? c.meses[mk] : null;
      if (md) {
        if (!footMeses[mk]) footMeses[mk] = { v: 0, pv: 0, h: 0, t: 0 };
        footMeses[mk].t += md.t;
        footMeses[mk].v += md.v;
        footMeses[mk].pv += md.pv;
        footMeses[mk].h += md.h;
      }
    }
  }
  var html = '<tfoot><tr>';
  html += '<td>TOTALES (' + cursos.length + ' cursos)</td>';
  html += '<td>' + totalPersonas + '</td>';
  html += '<td style="color:#fca5a5;">' + totalVenc + '</td>';
  html += '<td style="color:#fde68a;">' + totalPV + '</td>';
  for (var mj = 0; mj < monthKeys.length; mj++) {
    var mkF = monthKeys[mj];
    var fmd = footMeses[mkF];
    html += '<td>' + (fmd ? fmd.t : '-') + '</td>';
  }
  html += '</tr></tfoot>';
  return html;
}

// ── Filters ──────────────────────────────────────────────────
function _fccApplyFilters(cursos) {
  var result = cursos;
  if (_fccFilterCurso) {
    var q = _fccFilterCurso.toLowerCase();
    result = result.filter(function(c) {
      return (c.sigla || '').toLowerCase().indexOf(q) >= 0 || (c.nombre || '').toLowerCase().indexOf(q) >= 0;
    });
  }
  if (_fccFilterArea) {
    var areaUp = _fccFilterArea.toUpperCase();
    result = result.filter(function(c) {
      if (!c.parts) return false;
      for (var pi = 0; pi < c.parts.length; pi++) {
        if ((c.parts[pi].a || '').toUpperCase() === areaUp) return true;
      }
      return false;
    });
  }
  if (_fccFilterStatus) {
    if (_fccFilterStatus === 'VENCIDOS') result = result.filter(function(c) { return c.vencidos > 0; });
    else if (_fccFilterStatus === 'POR_VENCER') result = result.filter(function(c) { return c.porVencer > 0; });
    else if (_fccFilterStatus === 'AL_DIA') result = result.filter(function(c) { return c.pctAlDia === 100; });
  }
  return result;
}

// ── Sort ────────────────────────────────────────────────────
function _fccSort(col) {
  if (_fccSortCol === col) _fccSortAsc = !_fccSortAsc;
  else { _fccSortCol = col; _fccSortAsc = (col === 'sigla' || col === 'nombre'); }
  if (_fccData && _fccData.cursos) {
    _fccData.cursos.sort(function(a, b) {
      var va = (col === 'sigla') ? (a.nombre || a.sigla) : a[col];
      var vb = (col === 'sigla') ? (b.nombre || b.sigla) : b[col];
      if (typeof va === 'string') { va = va.toLowerCase(); vb = (vb || '').toLowerCase(); }
      if (va < vb) return _fccSortAsc ? -1 : 1;
      if (va > vb) return _fccSortAsc ? 1 : -1;
      return 0;
    });
    _fccRerender();
  }
}

// ── Export ───────────────────────────────────────────────────
function exportarFCCursos(formato) {
  if (!_fccData || !_fccData.cursos) { appAlert('No hay datos para exportar'); return; }
  var esPdf = (formato === 'pdf');
  var btn = document.querySelector('[data-testid="fcc-export-btn"]');
  var btnPdf = document.querySelector('[data-testid="fcc-export-pdf-global"]');
  var activeBtn = esPdf ? btnPdf : btn;
  if (activeBtn) { activeBtn.disabled = true; activeBtn.innerHTML = '<i class="ph ph-spinner"></i> ' + (esPdf ? 'Generando PDF...' : 'Exportando...'); }
  var handler = function(result) {
    if (activeBtn) { activeBtn.disabled = false; activeBtn.innerHTML = esPdf ? '<i class="ph ph-file-pdf"></i> PDF' : '<i class="ph ph-file-xls"></i> Excel'; }
    if (result && result.success) window.open(result.url, '_blank');
    else appAlert('Error: ' + (result && result.error || 'Error desconocido', 'err'));
  };
  var errHandler = function(err) {
    if (activeBtn) { activeBtn.disabled = false; activeBtn.innerHTML = esPdf ? '<i class="ph ph-file-pdf"></i> PDF' : '<i class="ph ph-file-xls"></i> Excel'; }
    appAlert('Error: ' + (err.message || err, 'err'));
  };
  if (esPdf) {
    google.script.run.withSuccessHandler(handler).withFailureHandler(errHandler).exportForecastCursosPDF(_fccData.cursos, _fccData.monthKeys);
  } else {
    google.script.run.withSuccessHandler(handler).withFailureHandler(errHandler).exportForecastCursosExcel(_fccData.cursos, _fccData.monthKeys);
  }
}

  for (var i = 0; i < _fccData.cursos.length; i++) {
